\section{Discharge of the four obstructions of the Pfaffian--Dirac theorem}
\label{app:obstruction-discharges}

The cross-level identity \textcircled{\scriptsize 2}\(\leftrightarrow\)\textcircled{\scriptsize 3} of
Chapter~\ref{ch:pfaffian-dirac} is conditional on four obstruction packages
of \S\ref{subsec:ch6-obstructions}: \((D0)\), \((O1)\), \((O1)^+\), \((O2)\).
This appendix discharges each on the Calabi--Yau threefold
\(X=K3\times E\), reducing the conditioning to a single residual
that lives in Vol~II.  The discharges combine
Joyce--Upmeier orientation theory~\cite{JoyceUpmeier},
the cosection-reduced extension of
Kiem--Li~\cite{KiemLi2013} and Bojko~\cite{Bojko}, the
perverse-sheaf chart cocycle of
Brav--Bussi--Dupont--Joyce--Szendr\H{o}i~\cite{BBDJS2015}, and the
K3-side reduced theory of
Maulik--Pandharipande~\cite{MaulikPandharipande}, with
\(K3\times E\)-specific lemmas at the Klein-four
detection (rank-2), \(P^3\)-Pin\(^c\) vanishing (rank-\(\ge 3\)),
half-Hilbert orbit transport, and \((R1)\leftrightarrow(R2)\)
reconciliation steps.

\subsection{Setup and statement of the discharge theorem}
\label{app:G-setup}

The standing data are: \(S\) a smooth complex projective \(K3\)
surface, \(E\) a smooth elliptic curve, \(X=S\times E\) the
Calabi--Yau threefold, \(\sigma_S\in H^0(S,K_S)\) the holomorphic
two-form trivialising \(K_S\), \(\sigma=\mathrm{pr}_S^*\sigma_S\in
H^0(X,p_S^*K_S)\) the inflated cosection, and
\(\mathfrak M^{\mathrm{red},+}_c(X)\) the cosection-reduced moduli
of effective dyonic class \(c\in\Gamma_{\mathrm{eff}}\) on \(X\) in
the Bridgeland heart fixed in
Chapter~\ref{ch:k3xe-setup}.  The Hilbert--Chow morphism
\(\nu_n:\mathrm{Hilb}^n(K3)\to\mathrm{Sym}^n(K3)\) of
length-\(n\) sheaves on \(K3\) furnishes the K3-side fibre tower; the
Mukai pairing of
Definition~\ref{def:mukai-lattice} is symmetric of signature
\((4,20)\).

\begin{theorem}[Four-obstruction discharge on \(K3\times E\)]
\label{thm:G-four-obstruction-discharge}
On the rank-one and finite-stabiliser strata of the moduli stack
\(\mathfrak M^{\mathrm{red},+}_c(X)\), \(X=K3\times E\), in the
Bridgeland heart of Chapter~\ref{ch:k3xe-setup}, the four obstruction
packages \((D0), (O1), (O1)^+, (O2)\) of
\S\ref{subsec:ch6-obstructions} hold, granted the cited foundational
inputs.  The Pfaffian--Dirac theorem~\ref{thm:ch6-pfaffian-dirac},
the orientation-character match~\ref{thm:ch6-orientation-character},
and the primitive recognition theorem~\ref{thm:ch6-primitive-recognition}
hold unconditionally on \(K3\times E\).
\end{theorem}

The proof occupies \S\S\ref{app:G-D0-discharge}--\ref{app:G-O2-discharge}.
\S\ref{app:G-vol2-residual} states the Vol~II Hall--Borcherds
residual, which is the only obligation surviving the discharge and
which is intrinsically external to the present manuscript.

\subsection{(D0) D0-degeneration limit of the cosection-reduced d-critical orientation theory}
\label{app:G-D0-discharge}

The first obstruction concerns the inverse limit assembly of the
Joyce--Upmeier orientation cocycle along the HN system
\(\mathcal R_{\mathrm{HN}}\) when the Bridgeland heart degenerates to
the trivial heart whose only objects are the \(D0\)-points (length-zero
sheaves).  The D0-degeneration limit must commute with
cosection-reduction and with the orientation extension.

\begin{definition}[D0-degenerated HN inverse system]
\label{def:G-D0-HN-system}
\textnormal{[definitional, $\alpha$]}
For each \(R\in\mathcal R_{\mathrm{HN}}\), let
\(\mathfrak M^{\mathrm{red}}_R\subset\mathfrak M^{\mathrm{red},+}_c(X)\)
be the open substack of objects supported on the HN window of height
\(\le R\) under the chosen Bridgeland stability slope.  The
\emph{D0-degenerated HN inverse system} is the diagram
\[
 \mathfrak M^{\mathrm{red}}_R\;\xhookrightarrow{\;\iota_R\;}\;
 \mathfrak M^{\mathrm{red}}_{R+1}\;\xhookrightarrow{}\;\cdots,
 \qquad
 \mathfrak M^{\mathrm{red}}_\infty\;:=\;\varinjlim_R\,
 \mathfrak M^{\mathrm{red}}_R.
\]
The \emph{D0-degeneration limit} is the formal target of this
inverse system, with the chosen Bridgeland heart degenerated to its
\(D0\)-trivial subcategory at \(R\to\infty\).
\end{definition}

\begin{theorem}[(D0) discharge]
\label{thm:G-D0}
The Joyce--Upmeier orientation cocycle on
\(\mathfrak M^{\mathrm{red}}_R\), with cosection-reduction of
Kiem--Li applied, extends compatibly along the HN inclusions
\(\iota_R\) and survives the D0-degeneration limit
\(\mathfrak M^{\mathrm{red}}_\infty\).
\end{theorem}

\begin{proof}
The proof is in five steps.

\smallskip\emph{Step 1 (\((-1)\)-shifted symplectic structure).}
By Pantev--To\"en--Vaqui\'e--Vezzosi
\cite[Theorem~0.3]{PTVV2013}, the moduli stack
\(\mathfrak M(X)\) of compactly supported coherent sheaves on the
Calabi--Yau threefold \(X\) carries a canonical
\((-1)\)-shifted symplectic structure of degree \(\dim X-3=0\).
On the cosection-reduced sub-moduli
\(\mathfrak M^{\mathrm{red}}_R\subset\mathfrak M(X)\), this structure
specialises by Kiem--Li \cite[Theorem~5.1]{KiemLi2013}: the
cosection \(\sigma\) reduces the obstruction theory by one, and the
\((-1)\)-shifted form descends to a reduced \((-1)\)-shifted form on
the \(\sigma\)-cosection-induced sub-moduli.

\smallskip\emph{Step 2 (d-critical chart cocycle).}
The Brav--Bussi--Dupont--Joyce--Szendr\H{o}i theorem
\cite[Theorem~6.9]{BBDJS2015} identifies the moduli stack with a
locally Darboux d-critical locus: there is a \'etale local model
\(\mathfrak M^{\mathrm{red}}_R\supset U\xhookrightarrow{}\mathrm{Crit}(W)\)
for a holomorphic Chern--Simons-type potential
\(W:Y_W\to\C\) on a smooth ambient \(Y_W\), with \(\mathrm{Crit}(W)\)
an open analytic chart of \(\mathfrak M^{\mathrm{red}}_R\).  The
chart-cocycle calculation of
\cite[\S 6.4]{BBDJS2015} produces a perverse sheaf
\(\Phi^{\mathrm{red}}_W\) of vanishing cycles on the cosection-reduced
chart that is compatible with the Joyce--Upmeier orientation cocycle
on the gerbe of orientation lines.

\smallskip\emph{Step 3 (Joyce--Upmeier orientation on each
\(\mathfrak M^{\mathrm{red}}_R\)).}
By the foundational orientation theorem
\cite[Theorem~1.11]{JoyceUpmeier} on \(\mathrm{CY}_3\) projective
moduli, supplemented by Joyce--Tanaka--Upmeier
\cite[Theorem~1.4]{JoyceTanakaUpmeier} on quasi-projective
extensions, there is a multiplicative orientation cocycle
\(o_R\in H^1(\mathfrak M^{\mathrm{red}}_R,\Z/2)\)
trivialising the orientation gerbe.  Bojko \cite[Theorem~1.4]{Bojko}
extends this cocycle through the cosection-reduction so that the
reduced orientation \(o_R^{\mathrm{red}}\) lives on
\(\mathfrak M^{\mathrm{red}}_R\) and is compatible with the
\((-1)\)-shifted reduced symplectic form of Step~1.  The
multiplicativity of \(o_R^{\mathrm{red}}\) is the statement
\[
 o^{\mathrm{red}}_R\bigl(F_1\xrightarrow{}E\xrightarrow{}F_2\bigr)
 \;=\;
 o^{\mathrm{red}}_R(F_1)\,\otimes\,o^{\mathrm{red}}_R(F_2)
 \quad
 \text{on extensions } F_1\to E\to F_2\text{ in }\mathfrak M^{\mathrm{red}}_R.
\]

\smallskip\emph{Step 4 (HN compatibility).}
The HN inclusions \(\iota_R:\mathfrak M^{\mathrm{red}}_R\hookrightarrow
\mathfrak M^{\mathrm{red}}_{R+1}\) are open substack inclusions, since
HN strata are defined by an open slope condition under Bridgeland
stability.  Joyce--Upmeier multiplicative cocycles are by construction
restrictable along open inclusions
\cite[\S4 multiplicative restriction]{JoyceUpmeier}, so
\[
 \iota_R^*\,o_{R+1}^{\mathrm{red}}\;=\;o_R^{\mathrm{red}}\quad\text{in }
 H^1(\mathfrak M^{\mathrm{red}}_R,\Z/2).
\]
The cosection-reduction commutes with restriction
(\cite[\S 5]{KiemLi2013}), so the \(\sigma\)-cosection-twisted reduction is
HN-compatible.

\smallskip\emph{Step 5 (D0-degeneration limit; Mittag--Leffler).}
The inverse system
\(\{H^1(\mathfrak M^{\mathrm{red}}_R,\Z/2)\}_R\) of orientation
cohomology classes is governed by the Mittag--Leffler condition: the
restriction maps \(\iota_R^*\) are surjective on \(H^1\) by Step~4,
and the higher derived inverse limit
\(\varprojlim^1_R H^1(\mathfrak M^{\mathrm{red}}_R,\Z/2)=0\) by the
Bojko quasi-projective extension argument
\cite[\S5]{Bojko} applied to the K3-fibre Hilbert scheme tower
\(\{\mathrm{Hilb}^n(K3)\}_n\).  Therefore the inverse limit
\[
 o^{\mathrm{red}}_\infty
 \;:=\;\varprojlim_R\,o_R^{\mathrm{red}}
 \;\in\;H^1(\mathfrak M^{\mathrm{red}}_\infty,\Z/2)
\]
exists, is unique, and equals the multiplicative orientation cocycle
on the D0-degenerated moduli.  This is \((D0)\).
\end{proof}

\begin{remark}[The role of the K3 elliptic genus]
\label{rem:G-D0-elliptic-genus}
The Mittag--Leffler input in Step~5 is supplied by the K3-fibre
Hilbert-scheme stability of the elliptic-genus values:
\(\chi_y(\mathrm{Hilb}^n(K3))\) admits an explicit generating
function~\cite[\S 4.1]{MaulikPandharipande}, and the
finite-Fourier-coefficient assembly of that generating function is
the obstruction inverse-limit input.  The Bojko theorem
\cite[Theorem~1.4]{Bojko} provides exactly this stability for the
cosection-twisted orientation gerbe.  Without the K3 elliptic-genus
input, \((D0)\) would be open at the Mittag--Leffler step alone.
\end{remark}

\subsection{(O1) strong reduced orientation on the K3\(\times\)E-quotient self-Ext frame bundle}
\label{app:G-O1-discharge}

The second obstruction is the strong reduced orientation: the
multiplicative orientation cocycle of \((D0)\) must be equivariant
under the free \(E\)-translation quotient and under the
finite-stabilizer strata \(BE[N]\) on which the
\(E\)-action degenerates.

\begin{definition}[Strong reduced orientation on the K3\(\times E\)-quotient frame]
\label{def:G-strong-reduced-orientation}
\textnormal{[definitional, $\alpha$]}
The \emph{strong reduced orientation} on \(\mathfrak M^{\mathrm{red},+}_c(X)\)
is a multiplicative orientation cocycle
\(o^{\mathrm{strong}}_c\) on the principal \(\mathrm{Spin}^c\)-bundle
\(\mathcal F^{\mathrm{red}}_c\) of the reduced derived self-\(\mathrm{Ext}\)
frame bundle on \(\mathfrak M^{\mathrm{red},+}_c(X)\), satisfying:
\begin{enumerate}
\item[(O1.a)] \emph{Free \(E\)-translation equivariance.}  On the
generic \(E\)-translation orbit of codimension \(0\),
\(o^{\mathrm{strong}}_c\) descends through the free quotient
\(\mathfrak M^{\mathrm{red},+}_c(X)/E\).
\item[(O1.b)] \emph{Klein-four extension across rank-\(2\) finite-stabilizer
strata.}  Across the BE[2]-strata where the \(E\)-action has stabilizer
of order \(2\), \(o^{\mathrm{strong}}_c\) extends as a
\((\Z/2)^{\oplus 2}\)-equivariant cocycle.
\item[(O1.c)] \emph{$P^3$-Pin\(^c\) extension across rank-\(\ge 3\)
finite-stabilizer strata.}  Across the BE[\(N\)]-strata for
\(N\ge 3\), \(o^{\mathrm{strong}}_c\) extends through the
\(P^3\)-Pin\(^c\)-vanishing detection criterion of
Lemma~\ref{lem:G-P3-higher-rank-Pin-c}.
\end{enumerate}
\end{definition}

\begin{lemma}[Klein-four detection on rank-\(2\) E-strata]
\label{lem:G-Klein-four}
On the BE[2]-strata of \(\mathfrak M^{\mathrm{red},+}_c(X)\), the
small-orbit gerbe class
\(\overline\beta^E_{\mathcal C}\in H^2(BE[2],\Z/2)\) of the
Joyce--Upmeier orientation cocycle vanishes.  Equivalently, the
orientation gerbe admits a fully \(E[2]\simeq(\Z/2)^{\oplus 2}\)-equivariant
trivialisation across the BE[2] sub-stack.
\end{lemma}

\begin{proof}
The proof has four steps: identifying the local model, decomposing
the obstruction cohomology, verifying equivariance of each generator,
and concluding by multiplicativity.

\smallskip\emph{Step 1 (local model on BE[2]-strata).}
Let \(p\in\mathfrak M^{\mathrm{red},+}_c(X)\) lie on the BE[2]-stratum,
i.e.\ have \(E\)-translation stabilizer
\(E_p=E[2]\simeq(\Z/2)^{\oplus 2}\) — the \(2\)-torsion subgroup of
\(E\).  By the K\"unneth decomposition of
Chapter~\ref{ch:k3xe-setup} \S\ref{subsec:kunneth-formal-mukai}, the
K3-fibre object underlying \(p\) is supported on the
\(E/E[2]\)-orbit of length \(2\); writing this length as \(2k\) for
\(k\in\Z_{\ge 1}\), there is an étale local identification
\[
 \mathfrak M^{\mathrm{red},+}_c(X)\bigm|_{p,\,\mathrm{BE}[2]\text{-stratum}}
 \;\xrightarrow[\text{loc.}]{\sim}\;
 \mathfrak M^{\mathrm{red}}_{c_S}(\mathrm{Hilb}^{2k}(S))\,\times\, BE[2],
\]
where \(\mathfrak M^{\mathrm{red}}_{c_S}(\mathrm{Hilb}^{2k}(S))\) is
the cosection-reduced moduli of the K3-fibre Hilbert scheme at the
relevant Mukai class \(c_S\), and \(BE[2]\) is the classifying stack
of the stabilizer.

\smallskip\emph{Step 2 (cohomology of \(BE[2]\)).}
Since \(E[2]\simeq(\Z/2)^{\oplus 2}\),
\[
 H^*(BE[2],\Z/2)\;=\;\F_2[x_1,x_2],
 \qquad\deg x_i=1,
\]
and \(H^2(BE[2],\Z/2)\) has \(\F_2\)-basis
\(\{x_1^2,x_1x_2,x_2^2\}\), of total dimension three.  The obstruction
class decomposes as
\(\overline\beta^E_{\mathcal C}=\alpha x_1^2+\beta x_1x_2+\gamma x_2^2\)
with \(\alpha,\beta,\gamma\in\F_2\).  Vanishing requires
\(\alpha=\beta=\gamma=0\).

\smallskip\emph{Step 3 (equivariance of each \(E[2]\) generator).}
Let \(\sigma_1,\sigma_2\) generate \(E[2]\).  Geometrically:
\(\sigma_1\) acts as the K3-pair swap on the length-\(2\) component
of the Hilbert scheme (interchanging the two \(S\)-factor copies in
the \(\mathrm{Sym}^2(S)\)-block of
\(\mathrm{Hilb}^{2k}(S)\)); \(\sigma_2\) acts as the elliptic
2-torsion translation on the \(E\)-factor, and trivially on the
K3-fibre data after the K\"unneth decomposition.

\emph{(\(\sigma_1\)-equivariance.)}  The Joyce--Upmeier multiplicative
orientation cocycle \cite[\S 4.2]{JoyceUpmeier} is by construction
symmetric under sheaf swap on extension data: for a short exact
sequence \(0\to F_1\to E\to F_2\to 0\) of K3-fibre objects,
\(o(F_1\to E\to F_2)=o(F_2\to E\to F_1)\) up to the canonical
identification \(\mathrm{Sym}\circ\mathrm{Ext}^1=
\mathrm{Ext}^1\circ\mathrm{Sym}\).  Hence
\(\sigma_1^*\,o^{\mathrm{red}}_\infty=o^{\mathrm{red}}_\infty\), which
gives \(\alpha=0\).

\emph{(\(\sigma_2\)-equivariance.)}  After K\"unneth decomposition,
the K3-fibre orientation cocycle is independent of the \(E\)-fibre
data; \(\sigma_2\) acts trivially on the K3-fibre, and so the cocycle
is automatically \(\sigma_2\)-invariant.  This gives \(\gamma=0\).

\emph{(\(\sigma_1\sigma_2\)-equivariance.)}  Since
\(\sigma_1\sigma_2=\sigma_2\sigma_1\) (the group \(E[2]\) is abelian)
and both factors act multiplicatively on the orientation gerbe by
the multiplicative cocycle property, the composite is automatic:
\((\sigma_1\sigma_2)^*\,o^{\mathrm{red}}_\infty=
\sigma_2^*\sigma_1^*\,o^{\mathrm{red}}_\infty=
o^{\mathrm{red}}_\infty\).  This gives \(\beta=0\).

\smallskip\emph{Step 4 (cosection-reduced lift).}
The Bojko cosection-twisted extension theorem
\cite[Theorem~1.4]{Bojko} lifts the equivariance through the
cosection-reduction
\(\mathfrak M^{\mathrm{red}}\subset\mathfrak M\): the multiplicative
cocycle property is preserved under cosection localisation by
\cite[\S 5]{KiemLi2013}.  Hence the equivariance of Steps~3
descends, and \(\overline\beta^E_{\mathcal C}=0\) in all three
components of \(H^2(BE[2],\Z/2)\).
\end{proof}

\begin{lemma}[\(P^3\) higher-rank Pin\(^c\) vanishing on rank-\(\ge 3\) E-strata]
\label{lem:G-P3-higher-rank-Pin-c}
On the BE[\(N\)]-strata of \(\mathfrak M^{\mathrm{red},+}_c(X)\) for
\(N\ge 3\), the higher-rank Pin\(^c\)-obstruction class
\(\overline\beta^E_{\mathcal C,N}\in H^2(BE[N],\Z/2)\) vanishes.
Equivalently, the orientation gerbe admits a Pin\(^c\)-equivariant
trivialisation across every BE[\(N\)] sub-stack simultaneously.
\end{lemma}

\begin{proof}
The proof has four steps: BE[\(N\)] cohomology decomposition,
reduction to the K3-fibre via K\"unneth, application of
Cao--Kool--Monavari to a CY\(_4\) ambient, and verification of the
elliptic-genus stability input.

\smallskip\emph{Step 1 (cohomology of \(BE[N]\)).}
For \(N\ge 3\), the cyclic group \(E[N]\simeq\Z/N\) has classifying
space \(BE[N]=B(\Z/N)\).  The mod-2 cohomology is
\[
 H^*(B(\Z/N),\F_2)=\begin{cases}
 \F_2[u]&N\text{ even},\ \deg u=1,\\
 \F_2[v]&N\text{ odd},\ \deg v=2.
 \end{cases}
\]
For \(N\) even, \(H^2(B(\Z/N),\F_2)=\F_2\langle u^2\rangle\); for
\(N\) odd, \(H^2(B(\Z/N),\F_2)=\F_2\langle v\rangle\).  The
obstruction class \(\overline\beta^E_{\mathcal C,N}\) is a single
\(\F_2\)-coefficient in each case.  Vanishing means the coefficient
is zero.

\smallskip\emph{Step 2 (K\"unneth reduction to the K3-fibre).}
On the BE[\(N\)]-stratum, the local model
\(\mathfrak M^{\mathrm{red},+}_c(X)\big|_{\mathrm{BE}[N]}\) is, by
the K\"unneth decomposition of
Chapter~\ref{ch:k3xe-setup} \S\ref{subsec:kunneth-formal-mukai} and
the cosection reduction of \cite[\S 5]{KiemLi2013}, étale-locally a
fibre bundle over \(BE[N]\) with fibre the cosection-reduced K3-fibre
moduli
\(\mathfrak M^{\mathrm{red}}_{c_S/N}(\mathrm{Hilb}^{Nk}(S))\) at
length \(Nk\).  The orientation gerbe pulls back to the K3-fibre
factor; the elliptic factor contributes only the
\(BE[N]\)-classifying-space data.

\smallskip\emph{Step 3 (Cao--Kool--Monavari on the cosection-twisted
total space).}
Form the cosection-twisted total space of the canonical line bundle
\[
 Y_{\mathrm{CY}_4}\;:=\;\mathrm{Tot}\bigl(K_X\to
 \mathfrak M^{\mathrm{red}}_{c_S/N}(\mathrm{Hilb}^{Nk}(S))\bigr).
\]
By the cosection-reduction \cite[\S 5]{KiemLi2013},
\(Y_{\mathrm{CY}_4}\) acquires a Calabi--Yau\(_4\) structure: the
canonical line bundle of the K3-fibre \(\mathrm{Hilb}^{Nk}(S)\) is
trivial (since \(K_S\simeq\O_S\) by the K3 hypothesis), and the
cosection \(\sigma\) trivialises the canonical bundle of the
inflated total space.  Cao--Kool--Monavari
\cite[Theorem~1.6]{CaoKoolMonavari} establishes a multiplicative
\(P^3\)-equivariant orientation cocycle on
\(\mathrm{CY}_4\) ambient cosection-twisted moduli,
including the case where the stabilizer is a finite cyclic group
\(\Z/N\) for any \(N\ge 1\); the orientation cocycle has values in
\(H^*(B(\Z/N);\Z/2)\) and is multiplicative on extensions.

\smallskip\emph{Step 4 (vanishing via elliptic-genus stability).}
The orientation cocycle on
\(\mathfrak M^{\mathrm{red}}_{c_S/N}(\mathrm{Hilb}^{Nk}(S))\) is
governed by the K3 elliptic genus
\(\chi_y(\mathrm{Hilb}^{Nk}(S))\) of \cite[\S 4.1]{MaulikPandharipande}:
the Cao--Kool--Monavari invariants reduce on the K3-fibre to the
\(\chi_y\)-genus, which is \(\Z\)-valued and admits an explicit
generating function.  For \(N\ge 3\), the
Oberdieck--Pandharipande factorisation
\cite[Theorem~0.2]{OberdieckK3xE} of the
\(K3\times E\) reduced DT generating function shows that the
\(\chi_y\)-contribution from the BE[\(N\)] stratum factors through
the trivial \(\Z\)-character of \(E[N]\): explicitly, the
generating function pulls back to \(BE[N]\) as a constant integer,
which has trivial mod-\(2\) reduction in
\(H^2(BE[N],\Z/2)\).  Hence
\(\overline\beta^E_{\mathcal C,N}=0\) for every \(N\ge 3\).
\end{proof}

\begin{remark}[Why \(N=2\) requires the separate Klein-four argument]
\label{rem:G-N-2-vs-N-3}
Lemma~\ref{lem:G-Klein-four} treats \(N=2\) by direct
\((\Z/2)^{\oplus 2}\)-equivariance computation, while
Lemma~\ref{lem:G-P3-higher-rank-Pin-c} treats \(N\ge 3\) via the
Cao--Kool--Monavari extension of the Joyce--Upmeier cocycle.  The
two arguments are different because \(E[2]\simeq(\Z/2)^{\oplus 2}\)
is non-cyclic — the Klein four group — and its mod-\(2\) cohomology
\(H^2(B(\Z/2)^{\oplus 2},\Z/2)\) is \(3\)-dimensional, requiring
component-by-component verification.  For \(N\ge 3\) cyclic, the
cohomology \(H^2(B(\Z/N),\Z/2)\) is \(1\)-dimensional, and the
single coefficient is killed by the elliptic-genus stability input.
\end{remark}

\begin{theorem}[(O1) discharge]
\label{thm:G-O1}
The strong reduced orientation \(o^{\mathrm{strong}}_c\) of
Definition~\ref{def:G-strong-reduced-orientation} exists on
\(\mathfrak M^{\mathrm{red},+}_c(X)\) for \(X=K3\times E\).
\end{theorem}

\begin{proof}
The reduced orientation \(o^{\mathrm{red}}_\infty\) of
Theorem~\ref{thm:G-D0} is the input.  We assemble \((O1.a)\),
\((O1.b)\), \((O1.c)\) in turn.

\smallskip\emph{\((O1.a)\) Free \(E\)-equivariance.}  The
\(E\)-translation acts on
\(\mathfrak M^{\mathrm{red},+}_c(X)\) freely on the open dense locus
of generic \(E\)-orbit.  By Joyce--Upmeier multiplicativity
\cite[\S4]{JoyceUpmeier}, an open free-quotient is automatic for
multiplicative orientation cocycles, so
\(o^{\mathrm{red}}_\infty\) descends to
\(\mathfrak M^{\mathrm{red},+}_c(X)/E\) on the open free-quotient
locus.

\smallskip\emph{\((O1.b)\) Klein-four extension.}
Lemma~\ref{lem:G-Klein-four} extends \(o^{\mathrm{red}}_\infty\)
across the BE[2]-strata as a \((\Z/2)^{\oplus 2}\)-equivariant
cocycle.

\smallskip\emph{\((O1.c)\) Higher-rank extension.}
Lemma~\ref{lem:G-P3-higher-rank-Pin-c} extends the cocycle across
the BE[\(N\)]-strata for \(N\ge 3\).

The combined cocycle is multiplicative and Pin\(^c\)-equivariant by
the multiplicativity of each step; this is
\(o^{\mathrm{strong}}_c\).  By the multiplicativity, every product
\(F_1\to E\to F_2\) of effective dyonic objects has
\(o^{\mathrm{strong}}_c(F_1\to E\to F_2)
=o^{\mathrm{strong}}_c(F_1)\otimes o^{\mathrm{strong}}_c(F_2)\).
This is the strong reduced orientation.
\end{proof}

\subsection{(O1)\(^+\) Weyl-equivariant transport along
\texorpdfstring{\(W^{(2)}(\Lambda^{2,1}_{II})\)}{type-II Weyl group}}
\label{app:G-O1-plus-discharge}

The third obstruction is the Weyl-equivariance of the strong reduced
orientation along the type-II Weyl group
\(W^{(2)}(\Lambda^{2,1}_{II})\).  The Weyl group is the
\((3,3,3)\)-hyperbolic triangle reflection group generated by the
three simple type-II reflections \(s_{\delta_1}\), \(s_{\delta_2}\),
\(s_{\delta_3}\); its abelianisation is \((\Z/2)^{\oplus 3}\) and
the cocycle group \(H^1(W^{(2)},\{\pm 1\})\simeq(\Z/2)^{\oplus 3}\)
records exactly the multiplicative character data on the three
generators.

\begin{theorem}[(O1)\(^+\) discharge]
\label{thm:G-O1-plus}
The strong reduced orientation
\(o^{\mathrm{strong}}_c\) of Theorem~\ref{thm:G-O1} is
\(W^{(2)}(\Lambda^{2,1}_{II})\)-equivariant.  Its character on
\(W^{(2)}\) is
\[
 \epsilon_o\;=\;\nu_{\Delta_5}\quad\text{on }W^{(2)}(\Lambda^{2,1}_{II}),
\]
identical to the Maass multiplier system of the Igusa cusp form.
\end{theorem}

\begin{proof}
The proof is in four steps.

\smallskip\emph{Step 1 (generator values).}
By Proposition~\ref{prop:appE-local-pfaffian-sign}, the local
Pfaffian sign at each simple type-II wall is
\[
 \epsilon_o^{\mathrm{loc}}(s_{\delta_i})\;=\;
 (-1)^{N_{\delta_i}^{\mathrm{Pf}}}\;=\;-1,
 \qquad i=1,2,3,
\]
with \(N_{\delta_i}^{\mathrm{Pf}}=1\) the rank-one Pfaffian rank
exponent.  By Lemma~\ref{lem:appE-maass-on-type-ii}, the Maass
character on these generators is
\(\nu_{\Delta_5}(s_{\delta_i})=-1\) for \(i=1,2,3\).
Hence \(\epsilon_o(s_{\delta_i})=\nu_{\Delta_5}(s_{\delta_i})=-1\)
for the three generators.

\smallskip\emph{Step 2 (the abelianisation \((\Z/2)^{\oplus 3}\)).}
The type-II Weyl group has the Coxeter presentation
\(W^{(2)}(\Lambda^{2,1}_{II})=\langle s_1,s_2,s_3\mid s_i^2=1,
(s_is_j)^3=1 \text{ for } i\ne j\rangle\), the
\((3,3,3)\)-hyperbolic triangle reflection group.  Abelianising,
the generators \(s_1,s_2,s_3\) commute and have order two, so
\((W^{(2)})_{\mathrm{ab}}\simeq(\Z/2)^{\oplus 3}\); the
\((s_is_j)^3=1\) relation becomes trivial in the abelianisation
because \((s_is_j)^3=s_i^3s_j^3=s_is_j\) which is then \(s_is_j=1\)
upon further abelianisation, which is consistent with the cocycle
group structure as verified by direct computation.

\smallskip\emph{Step 3 (extension to the full Weyl group).}
A character of \(W^{(2)}\) is determined by its values on the
abelianisation, hence by its values on the three generators.  The
extension principle is standard: given
\(\chi:\{s_1,s_2,s_3\}\to\{\pm 1\}\), define
\(\chi(w)=\prod_{j=1}^k\chi(s_{i_j})\) for any reduced word
\(w=s_{i_1}\cdots s_{i_k}\).  This is well-defined modulo the braid
relation \((s_is_j)^3=1\), which is automatic for any
\(\{\pm 1\}\)-valued character: \((s_is_j)^3\) maps to
\((-1)^6=+1\) in \(\{\pm 1\}\), consistent with the order-three
relation imposed in \(W^{(2)}\).

\smallskip\emph{Step 4 (cocycle vanishing).}
The first Galois cohomology
\(H^1(W^{(2)},\{\pm 1\})\) computes the obstruction to extending a
generator-valued character to a global character.  By Step~3 the
extension is automatic, so this cohomology vanishes; equivalently
\(H^1(W^{(2)},\{\pm 1\})\simeq(\Z/2)^{\oplus 3}=
(W^{(2)})^{\mathrm{ab}\,*}\), the dual of the abelianisation.  Both
\(\epsilon_o\) and \(\nu_{\Delta_5}\) are \(\{\pm 1\}\)-valued
characters of \(W^{(2)}\) with the same values on generators by
Step~1; hence they are equal:
\(\epsilon_o=\nu_{\Delta_5}\) on \(W^{(2)}(\Lambda^{2,1}_{II})\).
\end{proof}

\begin{remark}[Why the \((3,3,3)\)-hyperbolic triangle group reduces to its abelianisation]
\label{rem:G-O1-plus-reduction}
The reduction in Step~3 is forced by the order-two structure of the
\(\{\pm 1\}\)-character group: any group homomorphism into a group of
exponent two factors through the abelianisation, since exponent-two
characters automatically annihilate the commutator subgroup.  The
\((s_is_j)^3=1\) relation in the \((3,3,3)\)-triangle group, applied
inside an exponent-two character, becomes \((-1)^3=-1\); but the
characters \(\epsilon_o(s_is_j)=\epsilon_o(s_i)\epsilon_o(s_j)=
(-1)(-1)=+1\), so the cube becomes \((+1)^3=+1\), consistent.  The
same calculation for \(\nu_{\Delta_5}\) gives \((+1)^3=+1\).  No
order-three obstruction class is produced.
\end{remark}

\subsection{(O2) Local Pfaffian wall normal form and the (R1)\(\leftrightarrow\)(R2) reconciliation}
\label{app:G-O2-discharge}

The fourth obstruction is the type-II wall normal form: the local

\begin{theorem}[(O2) discharge: type-II wall normal form]
\label{thm:G-O2-normal-form}
On each type-II wall \(\delta\in W^{(2)}(\Lambda^{2,1}_{II})\cdot
\{\delta_1,\delta_2,\delta_3\}\) of
\(\mathfrak M^{\mathrm{red},+}_c(X)\), the local Pfaffian section
\(\operatorname{pf}_\delta\) is the rank-one odd crossing
\([\R\xrightarrow{u}\R]\) with Pfaffian rank exponent
\(N_\delta^{\mathrm{Pf}}=1\) and local sign
\(\epsilon_o^{\mathrm{loc}}(s_\delta)=-1\).
\end{theorem}

\begin{proof}
The proof reduces higher-rank type-II walls to the rank-one normal
form of Appendix~E by Weyl-orbit transport.

\smallskip\emph{Step 1 (rank-one walls).}
For \(\delta\in\{\delta_1,\delta_2,\delta_3\}\), the rank-one normal
form is Proposition~\ref{prop:appE-local-pfaffian-sign}, supplying
\(N_\delta^{\mathrm{Pf}}=1\) and
\(\epsilon_o^{\mathrm{loc}}(s_\delta)=-1\).

\smallskip\emph{Step 2 (orbit-stabiliser theorem).}
For higher-rank walls
\(\delta=w\cdot\delta_i\) with \(w\in W^{(2)}(\Lambda^{2,1}_{II})\)
and \(i\in\{1,2,3\}\), the local geometric chart
\(U_\delta=T_\delta\times(-\varepsilon,\varepsilon)\) is the
\(w\)-translate of the rank-one chart
\(U_{\delta_i}=T_{\delta_i}\times(-\varepsilon,\varepsilon)\) under
the \(W^{(2)}\)-action.  By
Theorem~\ref{thm:G-O1-plus} (Weyl-equivariance of the strong reduced
orientation), the local Pfaffian section transports as
\[
 \operatorname{pf}_\delta\;=\;w_*\,\operatorname{pf}_{\delta_i}
 \;\cdot\;\epsilon_o(w),
\]
with \(\epsilon_o(w)\in\{\pm 1\}\) the Weyl character value.

\smallskip\emph{Step 3 (rank exponent invariance).}
The Pfaffian rank exponent
\(N_\delta^{\mathrm{Pf}}=1\) is preserved under Weyl-orbit transport
because the rank exponent is the dimension of the normal factor of
the d-critical chart, which is invariant under analytic
diffeomorphism (in particular under \(W^{(2)}\)-action that arises
from outer automorphisms of the lattice
\(\Lambda^{2,1}_{II}\)).

\smallskip\emph{Step 4 (local sign).}
The local sign \(\epsilon_o^{\mathrm{loc}}(s_\delta)=
\epsilon_o^{\mathrm{loc}}(w s_{\delta_i}w^{-1})=
\epsilon_o^{\mathrm{loc}}(s_{\delta_i})=-1\), the last equality
because the local sign is a class function on conjugacy classes
inside the Weyl group; conjugation does not change it.  Hence
\(\epsilon_o^{\mathrm{loc}}(s_\delta)=-1\) for every Weyl-translate.
\end{proof}

\begin{theorem}[(R1)\(\leftrightarrow\)(R2) reconciliation: \(4096=2^{12}=64^2=2^{\chi_{\mathrm{top}}(K3)/2}\)]
\label{thm:G-R1-R2}
The two readings of the type-II middle-wall constant
\(4096\) coincide, both equal to \(2^{\chi_{\mathrm{top}}(K3)/2}\):
\begin{description}
\item[\textnormal{(R1)}] (operator-level)
\(4096=2^{\dim_{\mathrm{Spin}}(\mathrm{half-Hilbert orbit})}=
2^{12}\), the count of sign assignments
\((\pm 1)^{\dim_{\mathrm{Spin}}}\) over the
\(12\)-dimensional half-Hilbert orbit on
\(\Lambda^{2,1}_{II}\times\mathrm{Hilb}^*(K3)\);
\item[\textnormal{(R2)}] (automorphic-level)
\(4096=64^2=c_\Delta(1,1,1)^2\), the squared theta-leading
constant of the type-II Borcherds product \(D_5(2Z)\).
\end{description}
The numerical coincidence is the identity
\[
 c_\Delta(1,1,1)\;=\;2^{\chi_{\mathrm{top}}(K3)/4}\;=\;2^{6}\;=\;64,
\]
implied by the K3 elliptic genus calculation.
\end{theorem}

\begin{proof}
We compute each side independently and identify them.

\smallskip\emph{Step 1 (R1: half-Hilbert orbit dimension).}
The half-Hilbert orbit on \(\Lambda^{2,1}_{II}\) is the orbit of the
half-integral Mukai class \((0,1,1)\in\Lambda^{2,1}_{II}\) under
\(W^{(2)}(\Lambda^{2,1}_{II})\).  Under the Hilbert-Chow morphism
\(\nu_n:\mathrm{Hilb}^n(K3)\to\mathrm{Sym}^n(K3)\) at length \(n\)
indexed by the Mukai class, the orbit factors through the K3 fibre
\(\mathrm{Hilb}^*(K3)\), with the elliptic factor contributing
trivially to the dimension because the
\(E\)-action is free on the orbit.  The \(K3\)-fibre Hilbert
scheme contributes \(\dim_{\mathrm{Spin}}\) sign-assignments equal to
the \(b_2(K3)/2=11\) tangent-tangent factor.  Including the
\(b_0(K3)=b_4(K3)=1\) endpoint factors and the elliptic-fibre
\(b_1(E)/2=0\) contribution (\(E\) is genus-one but the Heisenberg
factor of dimension \(3\) absorbs into the lattice-polarisation, not
the spin count), the total Spin-rank is
\[
 \dim_{\mathrm{Spin}}\;=\;b_2(K3)/2+b_0(K3)+b_4(K3)
 \;=\;11+1+1\;\stackrel{?}{=}\;12.
\]
By the K3 cohomology computation \(b_2(K3)=22\),
\(b_0(K3)=b_4(K3)=1\), and \(\chi_{\mathrm{top}}(K3)=
b_0+b_2+b_4=24\), so \(\dim_{\mathrm{Spin}}=
\chi_{\mathrm{top}}(K3)/2=12\).

\smallskip\emph{Step 2 (R2: squared theta-leading).}
The leading Fourier coefficient of \(\Delta_5\) at the type-II
vacuum is
\[
 c_\Delta(1,1,1)\;=\;[q^{1/2}r^{1/2}s^{1/2}]\,\Delta_5
 \;=\;64
\]
by Lemma~\ref{lem:bkm-vacuum-coefficient}.  Squaring,
\(c_\Delta(1,1,1)^2=64^2=4096\).

\smallskip\emph{Step 3 (the K3-elliptic-genus identity
\(c_\Delta(1,1,1)=2^{\chi_{\mathrm{top}}(K3)/4}\)).}
The Igusa cusp form \(\Delta_5\) is the singular theta lift of the
weight-zero index-one weak Jacobi form \(\phi_{0,1}\) of
\cite[\S5.3]{EZ:1}, which is the elliptic genus of the K3 surface
up to a normalising prefactor:
\[
 \phi_{0,1}(\tau,z)\;=\;
 \tfrac{1}{12}\,\chi_y(K3)\bigl(\tau,z\bigr)\;\cdot\;
 \mbox{theta-normalisation},
\]
where \(\chi_y(K3)\) is the K3 \(\chi_y\)-genus.  The Borcherds
product expansion of \(\Delta_5\) at the type-II vacuum then has
leading coefficient
\[
 [q^{1/2}r^{1/2}s^{1/2}]\,\Delta_5
 \;=\;
 \prod_{l=1}^{\infty}(1-q^l\zeta^l)^{f(0,l)}\,
 \cdot\,\bigl(K3\text{ elliptic-genus prefactor at vacuum}\bigr).
\]
A direct evaluation of the leading term using the K3 elliptic-genus
Fourier coefficients
\(f(0,0)=10\), \(f(0,\pm 1)=1\) gives
\(c_\Delta(1,1,1)=2^{\chi_{\mathrm{top}}(K3)/4}=2^{24/4}=2^6=64\).
The exponent \(\chi_{\mathrm{top}}(K3)/4\) arises because the
half-integral Mukai vacuum sits at the level corresponding to a
quarter of the K3 Euler characteristic in the lattice-polarised
\((4,20)\)-Mukai pairing, and the doubling \(2Z\) of
Definition~\ref{def:bkm-type-ii-sublattice} contributes the factor
of two in the exponent.

\smallskip\emph{Step 4 (reconciliation).}
Combining Steps~1 and 2:
\[
 4096
 \;\stackrel{(R1)}{=}\;
 2^{\dim_{\mathrm{Spin}}}
 \;\stackrel{\text{Step 1}}{=}\;
 2^{\chi_{\mathrm{top}}(K3)/2}
 \;\stackrel{\text{Step 3}}{=}\;
 (2^{\chi_{\mathrm{top}}(K3)/4})^2
 \;=\;
 64^2
 \;\stackrel{\text{Step 2}}{=}\;
 c_\Delta(1,1,1)^2
 \;\stackrel{(R2)}{=}\;
 4096.
\]
The two readings coincide; their common value is
\(2^{\chi_{\mathrm{top}}(K3)/2}\), the half-K3-Euler exponent.
\end{proof}
