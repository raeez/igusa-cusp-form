\begin{definition}[Status labels for rows \(F_1\)--\(F_4\)]
Use the following labels for the full-paramodular rows.
\begin{description}
\item[\(\mathrm{CG\ product}\)] the Clery--Gritsenko
diagonal-divisor automorphic product, including group, character or
multiplier, divisor, and product normalization.
\item[\(\mathrm{GN\ denominator}\)] the Gritsenko--Nikulin
Weyl--Kac--Borcherds denominator algebra: root lattice, Weyl vector,
simple real roots, reflection character, product side, and Weyl-sum
side.
\item[\(\mathrm{CHL\ status}\)] a row map from CHL dyon denominator
data to the same Clery--Gritsenko row.  A CHL scalar square or a CHL
row of the same weight is not enough.
\item[\(\mathrm{physical\ host}\)] a compact or orbifold
Calabi--Yau/CHL/factorization object whose protected one-particle index
and reduced scalar partition function give the row in the same chamber
and cover.
\end{description}
The labels are logically independent.  A denominator theorem does not
construct a reduced curve-counting theory, and a scalar square does not
construct the denominator sign.
\end{definition}

\begin{theorem}[The \(K3\times E\)/OP/DT/\(\Delta_5\) chain]
\label{thm:f1-op-dt-delta5-chain}
Let \(S\) be an elliptically fibered \(K3\) surface with section, let
\(E\) be an elliptic curve, and put \(X=S\times E\).  For primitive
non-zero \(\beta_h\in H_2(S,\mathbb Z)\) with
\(\langle\beta_h,\beta_h\rangle=2h-2\), the quotient-by-\(E\),
Behrend-weighted Hilbert-scheme DT series satisfies, in the OP chamber,
\[
Z^X_{\mathrm{DT}}(T,Q,P)
=Z^X_{\mathrm{OP}}(P,Q,T)
=-4096\,\Delta_5(Z)^{-2},
\qquad(q,r,s)=(Q,P,T).
\]
Moreover
\[
\chi_{10}^{\mathrm{OP}}(P,Q,T)
=\left(64^{-1}\Delta_5(Z)\right)^2.
\]
Thus \(X=S\times E\) is a theorem-level reduced scalar host for
\(\Delta_5^{-2}\), and \(\Delta_5\) is the normalized automorphic square
root selected by the Borcherds product of \(\phi_{0,1}\).
\end{theorem}

\begin{proof}
Oberdieck--Pixton prove
\[
Z^X_{\mathrm{OP}}=-\left(\chi_{10}^{\mathrm{OP}}\right)^{-1}
\]
for the primitive reduced \(K3\times E\) partition function
\cite[Theorem 1]{OB}.  In their product normalization,
\(\chi_{10}^{\mathrm{OP}}=(64^{-1}\Delta_5)^2\); the local variable
and constant check is \(\texttt{main.tex:720--895}\) and
\(\texttt{agent\_material/10\_reduced\_dt\_pt\_op\_comparison\_theorem.tex}\).
Oberdieck's reduced DT/PT theorem for \(S\times E\) gives the
Hilbert-scheme DT to stable-pairs equality, and the
Oberdieck--Pandharipande primitive GW/PT comparison supplies the OP
series in the same variables \cite{OberdieckReducedPT,OPand}.  Since
\(\chi(S\times E)=24\cdot0=0\), the degree-zero MacMahon factor is
\(1\).  Combining these equalities gives the displayed scalar identity
and the constant \(-4096\).

The square-root identification is a separate automorphic theorem:
\[
\Delta_5=\operatorname{Borch}_{\theta}(\phi_{0,1}),\qquad
\phi_{0,1}=r^{-1}+10+r+O(q),
\]
so the Borcherds weight is \(f(0,0)/2=5\).  The half-K3 object used in
this lift is only a \(K_0\)-class
\(\mathbb U_{n,\ell}\) with
\(\operatorname{sdim}\mathbb U_{n,\ell}=f(n,\ell)\).  It is not a
canonical half of the \(K3\) Hilbert space.  Therefore the proved
physical content is the scalar OP/DT host plus the virtual determinant;
the compact microscopic \(E_3\)/Hall/chiral source is not constructed.
\end{proof}

\begin{proposition}[Full-paramodular rows \(F_2,F_3,F_4\)]
\label{prop:f2-f4-denominator-not-host}
Rows \(F_2,F_3,F_4\) have theorem-level Gritsenko--Nikulin denominator
algebras, but the audited sources do not construct a reduced
curve-counting, CHL, Hall, or factorization host for them.
Their product data are
\[
\begin{array}{c|c|c|c|c}
j&F_j&\phi_j&L_j&\rho_j\\ \hline
2&\Delta_2&\phi_{0,2}&M_2&(\frac14,\frac12,\frac14)\\
3&\Delta_1&\phi_{0,3}&M_3&(\frac16,\frac12,\frac16)\\
4&\Delta_{1/2}&\phi_{0,4}&M_4&(\frac18,\frac12,\frac18),
\end{array}
\]
with
\[
M_t=U(4t)\oplus\langle2\rangle,\qquad
e_t^{(n,\ell,m)}=q^nr^\ell s^{tm}.
\]
The denominator identities are
\[
\operatorname{den}(\mathfrak g^{\mathrm{GN}}_2)=\Delta_2,\qquad
\operatorname{den}(\mathfrak g^{\mathrm{GN}}_3)=\Delta_1,\qquad
\operatorname{den}(\mathfrak g^{\mathrm{GN}}_4)=\Delta_{1/2}.
\]
\end{proposition}

\begin{proof}
This is exactly Gritsenko--Nikulin Part II, Section 5.1.1, cases
\((t,II,\overline0)\), as transcribed in
\(\texttt{main.tex:3014--3225}\).  The product identities are
Gritsenko--Nikulin formulae (2.21), (2.20), and (2.11), and the
denominator theorem supplies the Weyl chamber, simple real roots, Weyl
vector, reflection character \(\det\), and Weyl-sum side \cite{GNII}.
This is stronger than a bare product formula.

The obstruction is physical, not automorphic.  No audited source gives,
for any of \(F_2,F_3,F_4\), an \(X_j\), a charge lattice, a reduced
obstruction theory, a protected one-particle state space, or a
partition theorem
\[
Z_j^{\mathrm{red}}=C_jF_j^{-2}.
\]
The CHL dyon denominator sources audited in
\(\texttt{agent\_material/11\_chl\_boundary\_primary\_source\_audit.tex}\)
map to \(F_5,\ldots,F_8\), not to these rows.  The formal current
envelope of \(\mathfrak g^{\mathrm{GN}}_t\) is a denominator target; it
does not reconstruct a compact Hall/factorization source.
\end{proof}

\begin{definition}[Physical row-map certificate for \(F_1\)--\(F_4\)]
\label{def:f1-f4-row-map-certificate}
To declare a physical host for a row \(F_j\), \(1\le j\le4\), one must
construct a certificate
\[
\mathfrak H_j=
\left(
X_j,\Gamma_j^{\mathrm{BPS}},\widetilde\Gamma_j,
\alpha_j^{\mathrm{phys}},\mathcal H^{1\mathrm p}_{j},
Z_j^{\mathrm{red}},\mathsf C_j,\Theta_j
\right)
\]
with the following properties.
\begin{enumerate}
\item \(X_j\) is a compact Calabi--Yau threefold, an oriented
Calabi--Yau orbifold, or a CHL background with a defined reduced
enumerative/protected index theory.
\item \(\widetilde\Gamma_j\) is the exact modular group or multiplier
cover on which \(F_j\) is a modular form with the character in
Theorem~\ref{thm:f1-f4-pinned-host-table}.
\item \(\Gamma_j^{\mathrm{BPS}}\) is the physical charge lattice and
\(\alpha_j^{\mathrm{phys}}:\Gamma_j^{\mathrm{BPS}}\to L_j\) matches
the product degree map:
\[
\alpha_j^{\mathrm{phys}}(\gamma(n,\ell,m))
=\alpha_j^{\mathrm{GN}}(n,\ell,m).
\]
\item The one-particle protected index has signed dimensions equal to
the Jacobi coefficients:
\[
\operatorname{sdim}\mathcal H^{1\mathrm p}_{1}(n,\ell)=f(n,\ell),
\qquad
\operatorname{sdim}\mathcal H^{1\mathrm p}_{j}(n,\ell)=f_j(n,\ell)
\quad(j=2,3,4),
\]
where \(\phi_{0,1}=\sum f(n,\ell)q^nr^\ell\) and
\(\phi_{0,j}=\sum f_j(n,\ell)q^nr^\ell\) for \(j=2,3,4\).
\item The reduced scalar partition function is proved, in the same
Fourier chamber and with the same orientation sign, to be
\[
Z_j^{\mathrm{red}}(Z)=C_jF_j(Z)^{-2}.
\]
\item The Hall, Drinfeld, or factorization object \(\mathsf C_j\)
has protected primitive bracket identified by \(\Theta_j\) with the
same denominator algebra:
\[
H^0\operatorname{Prim}_{\mathrm{prot}}(\mathsf C_j)
\cong\mathfrak g_j.
\]
\item The orientation character induced by the physical theory is the
automorphic/reflection character of the row.  For \(F_4\), the degree
\(8\) multiplier cover is part of the certificate, not a convention.
\end{enumerate}
A product formula, a denominator algebra, or a scalar square verifies
only the corresponding subset of this certificate.
\end{definition}
