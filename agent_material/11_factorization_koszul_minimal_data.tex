\subsection{Ambient categories}
\label{subsec:cell-eleven-ambient-categories}

Let \(X=K3\times E\), let
\(\Gamma=\Gamma_{\mathrm{BPS}}=\Z^3\), and let
\(\Gamma_{\mathrm{eff}}\subset\Gamma\) be the Igusa effective chamber.
All completions are finite-first:
$$
M=\varprojlim_{N,R} M_{N,R},
$$
where \(M_{N,R}\) is formed only after the product, coproduct,
descent maps, and transition maps have been constructed at the finite
Harder--Narasimhan truncation.  The raw cone is not enough; the finite
sets must come from a stability sector with HN finiteness and support
property.

The five data live in the following typed categories:
$$
\begin{array}{ccl}
\Fact^{\mathrm{ff}}_{\Ran(E),\Gamma}
&=&
\hbox{charge-completed factorization algebras on }\Ran(E),\\[2pt]
\Hall^{\mathrm{or,ff}}_{X,\sigma,o}
&=&
\hbox{oriented critical Hall factorization cosheaves},\\[2pt]
\Coalg^{\mathrm{conil,ff}}_{E,\Gamma}
&=&
\hbox{coaugmented conilpotent chiral coalgebras},\\[2pt]
\ChAlg^{\mathrm{ff}}_{E,\Gamma}
&=&
\hbox{charge-completed chiral algebras on }E,\\[2pt]
\LieSAlg_{\Gamma}
&=&
\hbox{charge-graded Lie superalgebras}.
\end{array}
$$
The bar-cobar functors are used only on their typed domain:
$$
\Omegach_E:
\Coalg^{\mathrm{conil,ff}}_{E,\Gamma}
\rightleftarrows
\ChAlg^{\mathrm{aug,ff}}_{E,\Gamma}:\barB_E^{\mathrm{ch}}.
$$
On the Koszul locus, the counit
$$
\varepsilon_{\mathcal A}:
\Omegach_E\barB_E^{\mathrm{ch}}(\mathcal A)
\longrightarrow
\mathcal A
$$
is a quasi-isomorphism.  This is an inversion theorem for a supplied
chiral algebra \(\mathcal A\).  It is not a construction of compact
observables.

\subsection{The five required objects}
\label{subsec:cell-eleven-five-objects}

\begin{definition}[The finite construction datum]
\label{def:cell-eleven-five-datum}
A compact factorization--Koszul realization of the Igusa current
envelope is a quintuple
$$
\mathfrak K_X
=
\left(
\mathcal F_X,\mathcal H_X,C_X,\Theta_{\mathrm{Kos}},
\Pi_{\mathrm{prim}}
\right),
\qquad X=K3\times E,
$$
with the following typed contents.

\smallskip
\noindent
\textbf{(F) Compact observable factorization algebra.}
\[
\mathcal F_X\in\Fact^{\mathrm{ff}}_{\Ran(E),\Gamma}.
\]
It is the \(E\)-pushforward of compact quantum observables of a BV
theory on \(X\), equipped with:
$$
\mathcal F_X=\varprojlim_{N,R}\mathcal F_{X,N,R},
\qquad
\mathcal F_{X,N,R}
=\bigoplus_{\gamma\in\Gamma_{\sigma}(N,R)}
\mathcal F_{X,\gamma},
$$
factorization products
$$
\mu^{\mathcal F}_{\gamma,\eta}:
\mathcal F_{X,\gamma}\boxtimes\mathcal F_{X,\eta}
\longrightarrow
\mathcal F_{X,\gamma+\eta}
$$
on disjoint configurations in \(\Ran(E)\), an augmentation
\(\epsilon_{\mathcal F}:\mathcal F_X\to\omega_E\), and a proof of
locality, descent, QME, anomaly cancellation, and finite-first
compatibility.

\smallskip
\noindent
\textbf{(H) Oriented compact Hall factorization object.}
\[
\mathcal H_X\in\Hall^{\mathrm{or,ff}}_{X,\sigma,o}.
\]
At finite stage it has the form
$$
\mathcal H_{X,N,R}
=
\bigoplus_{\gamma\in\Gamma_{\sigma}(N,R)}
H^{\BM}_{E,G_\gamma}
\left(
\mathfrak M_{\sigma,\gamma},
\van_\gamma\otimes\mathcal L_{o,\gamma}
\right),
$$
where \(\mathfrak M_{\sigma,\gamma}\) is a derived critical moduli
stack of compact objects, \(\mathcal L_{o,\gamma}\) is a strong
orientation line, and Hall multiplication is the pull-push
$$
\mu^{\mathcal H}_{\gamma,\eta}:
\mathcal H_{X,\gamma}\otimes\mathcal H_{X,\eta}
\xrightarrow{\,p^*,\,\mathrm{TS}_o,\,q_!\,}
\mathcal H_{X,\gamma+\eta}
$$
along extension correspondences
$$
\mathfrak M_{\sigma,\gamma}\times\mathfrak M_{\sigma,\eta}
\xleftarrow{\ p\ }
\mathfrak E_{\gamma,\eta}
\xrightarrow{\ q\ }
\mathfrak M_{\sigma,\gamma+\eta}.
$$
It also contains the three comparison maps
$$
\Theta^{\mathrm{or}}_{\hCS\to\Hall}:
\mathcal F_X\longrightarrow\mathcal H_X,
\qquad
\mathcal D_\hbar:
\mathcal H_X^+\widehat{\otimes}\mathcal H_X^-
\longrightarrow
\mathcal D_\hbar(\mathcal H_X),
$$
and
$$
\Psi^+_{\Hall\to\BKM}:
\Prim_{\mathrm{prot}}(\mathcal H_X)^+
\longrightarrow
\mathfrak n_+(\mathcal A_{\mathrm{den}}).
$$
The orientation character is required to be
$$
\epsilon_o=\nu_{\Delta_5}.
$$

\smallskip
\noindent
\textbf{(C) Chiral bar coalgebra.}
\[
C_X\in\Coalg^{\mathrm{conil,ff}}_{E,\Gamma}.
\]
It is supplied with a charge-preserving quasi-isomorphism of conilpotent
coalgebras
$$
b_X:
C_X\xrightarrow{\ \simeq\ }
\barB_E^{\mathrm{ch}}(\mathcal F_X),
$$
or, if the two-colour Vol.~II route is used,
$$
b_X:
C_X\xrightarrow{\ \simeq\ }
\barB_{\mathsf{SC}^{\mathrm{ch,top}}}
\left(
Z^{\mathrm{der}}_{\mathrm{ch}}(\mathcal F_X),\mathcal F_X
\right).
$$
The second line is legitimate only after the derived-centre pair is
constructed.  For a single chiral algebra, the promotion to the
two-colour \(\mathsf{SC}^{\mathrm{ch,top}}\)-algebra is additional
structure, not notation.

\smallskip
\noindent
\textbf{(Theta) Koszul comparison map.}
\[
\Theta_{\mathrm{Kos}}:
\Omegach_E C_X
\longrightarrow
\Den_{\Delta_5,E}
\]
is a morphism in \(\ChAlg^{\mathrm{ff}}_{E,\Gamma}\), equivalently a
twisting cochain
$$
\tau_{\mathrm{Kos}}:
C_X\longrightarrow \Den_{\Delta_5,E}[1],
$$
satisfying the chiral Maurer--Cartan equation.  It is required to be a
quasi-isomorphism after every finite truncation and after passage to the
finite-first inverse limit.

\smallskip
\noindent
\textbf{(Pi) Primitive Lie comparison.}
\[
\Pi_{\mathrm{prim}}:
H^0\Prim_{\mathrm{prot}}(\mathcal F_X)
\xrightarrow{\ \sim\ }
\mathcal A_{\mathrm{den}}
\]
is an isomorphism in \(\LieSAlg_{\Gamma}\), with components
$$
\Pi_{\mathrm{prim},\gamma}:
H^0\Prim_{\mathrm{prot}}(\mathcal F_X)_\gamma
\xrightarrow{\ \sim\ }
\mathcal A_{\mathrm{den},\alpha(\gamma)}.
$$
It must identify the Hall/factorization super-commutator with the
Borcherds bracket, including real Serre relations, imaginary simple-root
parities and multiplicities, orthogonality relations, and the radical
quotient in the completed Hall--Drinfeld double.
\end{definition}

\subsection{Compatibility diagrams}
\label{subsec:cell-eleven-diagrams}

The five objects are not independent decorations.  They must satisfy
the following diagrams.

\begin{enumerate}
\item \textbf{Finite-first transition compatibility.}  For
\((N,R)\leq(N',R')\),
$$
\begin{array}{ccc}
\Omegach_E C_{X,N',R'} & \xrightarrow{\Theta_{\mathrm{Kos},N',R'}}
& \Den_{\Delta_5,E,N',R'}\\
\downarrow && \downarrow\\
\Omegach_E C_{X,N,R} & \xrightarrow{\Theta_{\mathrm{Kos},N,R}}
& \Den_{\Delta_5,E,N,R}.
\end{array}
$$
The vertical arrows are transition maps in the completed category.
Quasi-isomorphism is first checked at finite stage and then passed to
the limit by the stated completeness argument.

\item \textbf{Bar-cobar compatibility with the compact source.}
$$
\begin{array}{ccc}
\Omegach_E C_X
& \xrightarrow{\Theta_{\mathrm{Kos}}}
& \Den_{\Delta_5,E}\\
\downarrow{\scriptstyle \Omegach_E b_X}
&&
\uparrow{\scriptstyle \Theta_{\mathcal F}}\\
\Omegach_E\barB_E^{\mathrm{ch}}(\mathcal F_X)
& \xrightarrow{\varepsilon_{\mathcal F}}
& \mathcal F_X
\end{array}
$$
Here \(\varepsilon_{\mathcal F}\) is the Vol.~I counit.  The right-hand
map \(\Theta_{\mathcal F}\) is additional realization data supplied as
part of the compact datum; it is not induced by the determinant or by
the formal current envelope.  Without this square,
\(\Theta_{\mathrm{Kos}}\) is merely a map from an abstract coalgebra to
the current envelope.

\item \textbf{Primitive bracket compatibility.}  For all
\(\gamma,\eta\in\Gamma_{\mathrm{eff}}\),
$$
\begin{array}{ccc}
H^0\Prim_{\mathrm{prot}}(\mathcal F_X)_\gamma
\otimes
H^0\Prim_{\mathrm{prot}}(\mathcal F_X)_\eta
& \xrightarrow{[-,-]_{\mathcal F}}
& H^0\Prim_{\mathrm{prot}}(\mathcal F_X)_{\gamma+\eta}\\
\downarrow{\scriptstyle \Pi_{\mathrm{prim},\gamma}\otimes
\Pi_{\mathrm{prim},\eta}}
&&
\downarrow{\scriptstyle \Pi_{\mathrm{prim},\gamma+\eta}}\\
\mathcal A_{\mathrm{den},\alpha(\gamma)}
\otimes
\mathcal A_{\mathrm{den},\alpha(\eta)}
& \xrightarrow{[-,-]_{\BKM}}
& \mathcal A_{\mathrm{den},\alpha(\gamma+\eta)}
\end{array}
$$
This square is the bracket theorem.  The equality of signed dimensions
does not imply it.

\item \textbf{Protected integration and denominator compatibility.}
$$
\begin{array}{ccc}
K_0 H^0\Prim_{\mathrm{prot}}(\mathcal F_X)_\gamma
& \xrightarrow{\chi^{\mathrm{prot}}}
& \Z\\
\downarrow{\scriptstyle K_0\Pi_{\mathrm{prim},\gamma}}
&&
\Vert\\
K_0\mathcal A_{\mathrm{den},\alpha(\gamma)}
& \xrightarrow{\sdim}
& \Z
\end{array}
$$
The common value must be
$$
\chi^{\mathrm{prot}}(\gamma)
=\sdim\mathcal A_{\mathrm{den},\alpha(\gamma)}
=f(nm,l),
\qquad \gamma=(n,l,m).
$$
Consequently the protected determinant is
$$
e^{-2\pi i(\rho,z)}
\prod_{\gamma\in\Gamma_{\mathrm{eff}}}
\left(1-e^{-2\pi i(\alpha(\gamma),z)}\right)^{
\chi^{\mathrm{prot}}(\gamma)}
=64^{-1}\Delta_5(2Z).
$$

\item \textbf{Hall--Drinfeld/BKM compatibility.}
The positive-half comparison must commute with Hall brackets:
$$
\begin{array}{ccc}
\Prim_{\mathrm{prot}}(\mathcal H_X)^+_\gamma
\otimes
\Prim_{\mathrm{prot}}(\mathcal H_X)^+_\eta
& \xrightarrow{[-,-]_{\Hall}}
& \Prim_{\mathrm{prot}}(\mathcal H_X)^+_{\gamma+\eta}\\
\downarrow{\scriptstyle \Psi^+_{\Hall\to\BKM}\otimes
\Psi^+_{\Hall\to\BKM}}
&&
\downarrow{\scriptstyle \Psi^+_{\Hall\to\BKM}}\\
\mathfrak n_{+,\alpha(\gamma)}(\mathcal A_{\mathrm{den}})
\otimes
\mathfrak n_{+,\alpha(\eta)}(\mathcal A_{\mathrm{den}})
& \xrightarrow{[-,-]_{\BKM}}
& \mathfrak n_{+,\alpha(\gamma+\eta)}(\mathcal A_{\mathrm{den}}).
\end{array}
$$
The completed pairing and radical quotient are required to make
\(\mathcal D_\hbar(\mathcal H_X)\) descend to the Borcherds double with
orientation character \(\nu_{\Delta_5}\).
\end{enumerate}

\subsection{Sufficiency theorem}
\label{subsec:cell-eleven-sufficiency}

\begin{theorem}[Finite dossier suffices]
\label{thm:cell-eleven-dossier-suffices}
If the five objects of Definition~\ref{def:cell-eleven-five-datum}
exist and satisfy the diagrams of
Section~\ref{subsec:cell-eleven-diagrams}, then
\(\Den_{\Delta_5,E}\) is the compact \(K3\times E\)
factorization--Koszul realization of the Igusa denominator algebra:
$$
\Omegach_E C_X\xrightarrow{\ \simeq\ }\Den_{\Delta_5,E}.
$$
Moreover,
$$
H^0\Prim_{\mathrm{prot}}(\mathcal F_X)
\cong
\mathfrak g_{\Delta_5}
$$
as charge-graded Lie superalgebras, and its denominator is
\(64^{-1}\Delta_5(2Z)\).
\end{theorem}

\begin{proof}
The first assertion is exactly the quasi-isomorphism
\(\Theta_{\mathrm{Kos}}\).  The primitive bracket square identifies the
protected primitive bracket with the Borcherds bracket of
\(\mathcal A_{\mathrm{den}}\).  The protected integration square gives
the signed multiplicity equality \(f(nm,l)\).  Substitution into the
Borcherds product gives \(64^{-1}\Delta_5(2Z)\).  No scalar identity is
used to manufacture the chain-level data; it enters only after the
primitive comparison has been proved.
\end{proof}

cancellation, charge completion, and augmentation.

For \(\mathcal H_X\), the local compact Hall notes construct a finite

HN finiteness, critical charts, strong orientation, extension
correspondences, and reduced or equivariant \(E\)-descent.  The OP
Hilbert-scheme scalar branch is proper and Behrend-weighted, but it is

Lie superalgebra and the signed multiplicities.  It does not construct
physical protected primitives or prove the Hall super-commutator
satisfies the Gritsenko--Nikulin/Borcherds defining relations.

Thus all five compact source/comparison blocks remain open.  Existing
material supplies a precise template, not a theorem.
\end{proof}

\subsection{Finite proof checklist}
\label{subsec:cell-eleven-checklist}

A proof of the compact factorization--Koszul theorem is finite.  It is
complete exactly when the following checks are complete.

\begin{enumerate}
\item Construct a stability sector \((\mathcal A_\sigma,\sigma,S)\) for
compact \(K3\times E\) objects with HN property, support property, and
bounded semistable substacks.
\item Construct finite HN truncations \(\Gamma_\sigma(N,R)\) and prove
transition compatibility.  Do not use raw finite lower subsets of
\(\Gamma_{\mathrm{eff}}\).
\item Construct \(\mathcal F_X\) as a charge-completed factorization
algebra on \(\Ran(E)\): BV fields, differential, quantum master
equation, anomaly cancellation, locality, descent, augmentation, and
finite-first completion.
\item Construct \(\mathcal H_X\): critical charts
\(\mathfrak M_{\sigma,\gamma}|_U\simeq[\Crit(f_{\gamma,U})/G_{\gamma,U}]\),
vanishing-cycle coefficients, strong orientation lines, Hall extension
correspondences, proper pull-push, Thom--Sebastiani transport, and
associativity over two-step flags.
\item Prove \(E\)-equivariant or reduced Hall descent.  In particular,
retain the relative \(E\)-position lost by the scalar quotient.
\item Prove the orientation character identity
\(\epsilon_o=\nu_{\Delta_5}\).
\item Construct
\(\Theta^{\mathrm{or}}_{\hCS\to\Hall}\), prove it preserves products,
coproducts, charge, orientation, and protected integration.
\item Construct the protected primitive functor
\(\Prim_{\mathrm{prot}}\), its super-commutator, and its charge
conservation.
\item Prove
\[
\sdim\Prim_{\mathrm{prot}}(\mathcal F_X)_{(n,l,m)}=f(nm,l)
\]

denominator.
\item Construct \(C_X\) in
\(\Coalg^{\mathrm{conil,ff}}_{E,\Gamma}\), prove conilpotence, and
place any curvature in the declared coderived/completed category.
\item Prove \(b_X:C_X\simeq\barB_E^{\mathrm{ch}}(\mathcal F_X)\), or
construct the two-colour
\(\mathsf{SC}^{\mathrm{ch,top}}\)-pair and its bar coalgebra.
\item Verify the Koszul-locus hypotheses for \(\mathcal F_X\) at every

\item Construct the twisting cochain

\item Prove
\(\Theta_{\mathrm{Kos}}:\Omegach_E C_X\to\Den_{\Delta_5,E}\) is a

\item Identify the three real simple protected primitives with

\item Prove the real Serre relations, imaginary simple-root parity and

\(\Pi_{\mathrm{prim}}\).  This is the non-scalar bracket theorem.
\item Only after the previous items, apply the Borcherds denominator
