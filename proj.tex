\begin{abstract}
Let $X=S\times E$, with $S$ a K3 surface and $E$ an elliptic curve.
Define the reduced D6--D2--D0 index by the Behrend-weighted Euler
characteristic of $\mathrm{Hilb}^n(X,(\beta_h,d))/E$.  Scalar-square
branch:
$$Z^X_{\square}=C/\Delta_5^2.$$
In the primitive reduced curve-counting chamber this is
Oberdieck--Pixton; in the Behrend-weighted Hilbert-scheme normalization
it is an assumption.
Set
$$\mathcal D_X^2=C/Z^X_{\square}.$$
Cusp branch:
$$\mathcal D_X=\Delta_5
=64q^{1/2}r^{1/2}s^{1/2}
\prod_{(n,l,m)>0}(1-q^nr^ls^m)^{f(nm,l)},$$
where
$$\phi_{0,1}=\frac12 Z_{\mathrm{K3}}
=\operatorname{Str}_{\mathcal H_{K3}}q^{L_0-c/24}r^{J_0}
=\sum f(n,l)q^nr^l.$$
$$\mathcal D_X^{-1}
=\frac{1}{64}q^{-1/2}r^{-1/2}s^{-1/2}\sch\mathcal F(\mathcal V),$$
$$\mathcal D_X
=64q^{1/2}r^{1/2}s^{1/2}\sdet_{\mathcal V}(1-x).$$
$$\frac1{64}\mathcal D_X(2Z)=\operatorname{den}(\autcor).$$
The lattice is $\La^{3,2}$; the chamber lies in
$\La^{2,1}_{II}$; the Weyl vector is
$$\rho=f_2-\frac12f_3+f_{-2}.$$
For
$$\alpha(n,l,m)=2nf_2-lf_3+2mf_{-2},$$
$$f(nm,l)=
\dim \mathfrak{g}_{\alpha(n,l,m),\overline 0}
-\dim \mathfrak{g}_{\alpha(n,l,m),\overline 1},\qquad
q^nr^ls^m=e^{-\pi i(\alpha(n,l,m),z)}.$$
\end{abstract}

\section{The one-particle index and its determinant}

Put
$$q=e^{2\pi iz_1},\qquad r=e^{2\pi iz_2},\qquad s=e^{2\pi iz_3}.$$

The one-particle input is the Ramond supertrace on the K3 sigma model:
$$Z_{\mathrm{K3}}(\tau,z)=
\operatorname{Tr}_{RR(\mathrm{K3})}
(-1)^{F_L+F_R}q^{L_0-c/24}r^{J_0},\qquad c=6.$$
Normalize:
$$Z_{\mathrm{K3}}(\tau,z)=2\phi_{0,1}(\tau,z),$$
where $\mathcal H_1$ carries the Ramond ground-state shifted grading and
$$\phi_{0,1}(\tau,z)
=\operatorname{Str}_{\mathcal{H}_1}q^{L_0}r^{J_0}
=\sum_{n\geq 0,\;l\in\Z}f(n,l)q^nr^l.$$
The virtual graded super vector space $\mathcal H_1$ is fixed by
$$\sdim(\mathcal H_{1;n,l})=f(n,l).$$

The $q^0$ term is
$$\phi_{0,1}=r^{-1}+10+r+O(q),$$
and Borcherds' weight is
$$\frac{f(0,0)}{2}=5.$$

\begin{lemma}[Fock trace and superdeterminant]
Let
$$V=\bigoplus_{\lambda\in P}V_\lambda$$
be a $P$-graded super vector space with finite-dimensional homogeneous
pieces and let $a_\lambda=\sdim V_\lambda$.  Put
$$\mathcal F(V)=\operatorname{Sym}(V_{\overline 0})
\otimes \bigwedge(V_{\overline 1}).$$
As a formal character,
$$\sch\mathcal F(V):=
\sum_{\mu\in P}\sdim\mathcal F(V)_\mu\,x^\mu
=\prod_{\lambda\in P}(1-x^\lambda)^{-a_\lambda}.$$
The inverse character is the superdeterminant
$$\sdet_V(1-x)=\prod_{\lambda\in P}(1-x^\lambda)^{a_\lambda}.$$
\end{lemma}

\begin{proof}
A one-dimensional even generator of weight $\lambda$ contributes
$1+x^\lambda+x^{2\lambda}+\cdots=(1-x^\lambda)^{-1}$.  A
one-dimensional odd generator contributes $1-x^\lambda$.  Multiplying
over a homogeneous basis proves the formula.
\end{proof}

Write
$$x^{(n,l,m)}=q^nr^ls^m.$$
Put
$$\mathcal V=\bigoplus_{(n,l,m)>0}\mathcal V_{n,l,m},\qquad
\sdim\mathcal V_{n,l,m}=f(nm,l).$$
DMVV second quantization \cite{DMVV}:
$$\sch\mathcal F(\mathcal V)
=\prod_{(n,l,m)>0}(1-x^{(n,l,m)})^{-f(nm,l)}.$$
Inverse determinant:
$$\sdet_{\mathcal V}(1-x)
=\prod_{(n,l,m)>0}(1-x^{(n,l,m)})^{f(nm,l)}.$$
Vacuum factor:
$$v_0(Z)=64q^{1/2}r^{1/2}s^{1/2}.$$
It is an exterior vacuum prefactor, not a homogeneous summand of
$\mathcal V$.

\begin{definition}[Chiral determinant]
The chiral determinant attached to the one-particle index is
$$\mathcal{D}_X(Z)
=v_0(Z)\sdet_{\mathcal V}(1-x)
=64q^{1/2}r^{1/2}s^{1/2}
\prod_{(n,l,m)>0}(1-q^nr^ls^m)^{f(nm,l)}.$$
Its inverse is the chiral BPS factor
$$\mathcal Z_{\mathrm{BPS}}^{\mathrm{ch}}(Z)
=\mathcal D_X(Z)^{-1}
=\frac{1}{64}q^{-1/2}r^{-1/2}s^{-1/2}
\prod_{(n,l,m)>0}(1-q^nr^ls^m)^{-f(nm,l)}.$$
\end{definition}

\begin{proposition}
The determinant constructed above has initial term
$$64q^{1/2}r^{1/2}s^{1/2},$$
and
$$\mathcal Z_{\mathrm{BPS}}^{\mathrm{ch}}
=\frac{1}{64}q^{-1/2}r^{-1/2}s^{-1/2}\sch\mathcal F(\mathcal V),$$
$$\mathcal D_X
=64q^{1/2}r^{1/2}s^{1/2}\sdet_{\mathcal V}(1-x).$$
\end{proposition}

\begin{proof}
Apply the Fock trace lemma to $\mathcal V$:
$$\sch\mathcal F(\mathcal V)
=\prod_{(n,l,m)>0}(1-q^nr^ls^m)^{-f(nm,l)},\qquad
\sdet_{\mathcal V}(1-x)
=\prod_{(n,l,m)>0}(1-q^nr^ls^m)^{f(nm,l)}.$$
Multiplication by $v_0(Z)$ gives $\mathcal D_X$; multiplication by
$v_0(Z)^{-1}$ gives $\mathcal Z_{\mathrm{BPS}}^{\mathrm{ch}}$.
\end{proof}

\begin{definition}[Root character map]
$$z=z_1f_{-2}+z_2f_3+z_3f_2,\qquad
\alpha(n,l,m)=2nf_2-lf_3+2mf_{-2}.$$
With the bilinear form on $\La^{2,1}_{II}$,
$$q^{1/2}r^{1/2}s^{1/2}=\exp(-\pi i(\rho,z)),$$
and
$$q^nr^ls^m=\exp(-\pi i(\alpha(n,l,m),z)).$$
Under $Z\mapsto2Z$:
$$\exp(-2\pi i(\rho,z))
\qquad\textrm{and}\qquad
\exp(-2\pi i(\alpha(n,l,m),z)).$$
Thus $\Delta_5(Z)$ is the half-character product and $\Delta_5(2Z)$ is
the integral denominator-character product.
\end{definition}

\begin{definition}[Root superdimension]
$$\smult(\alpha(n,l,m))
:=\dim \mathfrak{g}_{\alpha(n,l,m),\overline 0}
-\dim \mathfrak{g}_{\alpha(n,l,m),\overline 1}
=f(nm,l).$$
\end{definition}

\section{The normalized cusp form}

Freitag \cite{FREITAG:1} and Van der Geer \cite{VDGEER:1}:
$$\mathcal{SM}(\Sp_4(\Z)) = \C[E_4,E_6,\Delta_{10},\Delta_{12}],$$
with weights
$$\mathrm{wt}(E_4,E_6,\Delta_{10},\Delta_{12})=(4,6,10,12).$$
Square root:
$$\Delta_{10}=\Delta_5^2,\qquad
\mathrm{wt}(\Delta_5)=5,\qquad
\Delta_5|_5 g=\nu_{\Delta_5}(g)\Delta_5.$$
Theta-constant normalization:
$$\Delta_5 =
\prod_{\substack{(a,b)\in (\Z/2\Z)^2\\{}^t ab\equiv 0 \mod 2}}
\nu_{a,b},$$
with
$$\nu_{ab}(z)=
\sum_{l\in\Z^2}\exp\bigl(\pi i(z[l+\tfrac12a]+{}^tbl)\bigr),
\qquad z[x]={}^txzx.$$

Let $M = \begin{pmatrix} A & B\\
C & D\end{pmatrix} \in \Sp_4(\Z)$.  On the standard generators,
Maass's character \cite{MAASS:1}
$$\nu_{\Delta_5}:\Sp_4(\Z)\rightarrow \C^\times,\qquad
|\nu_{\Delta_5}|=1,$$
is
$$\nu_{\Delta_5}\begin{pmatrix}0 & \Id_2\\
-\Id_2 & 0\end{pmatrix} = 1,$$ $$\nu_{\Delta_5}\begin{pmatrix}\Id_2 & B\\
0 & \Id_2\end{pmatrix} = (-1)^{b_1 + b_2 + b_3},$$
$$\nu_{\Delta_5}\begin{pmatrix}{}^tA^{-1} & 0\\
0 & A\end{pmatrix} = (-1)^{(1 + a_1 + a_4)(1 + a_2 + a_3) + a_1a_4}$$
where $A = \begin{pmatrix}a_1 & a_2\\
a_3 & a_4\end{pmatrix} \in \GL_2(\Z)$ and $B = \begin{pmatrix}b_1 &
b_2\\
b_2 & b_3\end{pmatrix} \in M_{2\times2}(\Z).$

$\Delta_5$ is a weight-$5$ form with character
$\nu_{\Delta_5}$, and $\Delta_5^2=\Delta_{10}$ is scalar.

For
$$Z=\begin{pmatrix}z_1&z_2\\z_2&z_3\end{pmatrix},$$
$$\Delta_5(Z) =
\sum_{\substack{n,l,m\equiv 1 \bmod 2\\4nm-l^2>0\\n,m>0}}
c_\Delta(n,l,m)\exp\bigl(\pi i(nz_1+lz_2+mz_3)\bigr),$$
with
$$c_\Delta(1,1,1)=64,\qquad c_\Delta(n,l,m)\in64\Z.$$
The isotropic cusp identity is
$$1+\frac1{64}\sum_{t\geq1}c_\Delta(1+2t,1,1)q^t
=\prod_{k\geq1}(1-q^k)^9.$$

\begin{proof}[Cusp computation]
Fourier--Jacobi expansion:
$$\Delta_5(Z) =
\sum_{\substack{m>0\\m\equiv1\bmod2}}
\phi_{5,m}(z_1,z_2)\exp(\pi i m z_3).$$
The first coefficient has weight $5$, index $\frac12$, and the
non-trivial character.  Put
$$\nu_{11}(z_1,z_2)=
\sum_{n\in\Z}(-1)^n
\exp\left(\frac{\pi i}{4}(2n+1)^2z_1+\pi i(2n+1)z_2\right).$$
Jacobi triple product:
$$\nu_{11} =
q^{1/8}r^{-1/2}\prod_{n\geq1}
(1-q^{n-1}r)(1-q^nr^{-1})(1-q^n),$$
where $q=\exp(2\pi iz_1)$ and $r=\exp(2\pi iz_2)$.
Set
$$\psi_{5,\frac12}=\eta(z_1)^9\nu_{11}(z_1,z_2),\qquad
\eta(z_1)=q^{1/24}\prod_{n\geq1}(1-q^n).$$
Squared form: weight $10$, index $1$.  The one-dimensional
space contains the first Fourier--Jacobi coefficient of
$$\Delta_{10}=\Delta_5^2.$$
Coefficient comparison:
$$\frac1{64}\phi_{5,1}(z_1,z_2)=\psi_{5,\frac12}(z_1,z_2)
=-q^{1/2}r^{-1/2}\prod_{n\geq1}
(1-q^{n-1}r)(1-q^nr^{-1})(1-q^n)^{10}.$$
Coefficient of $r^{1/2}$:
$$1+\frac1{64}\sum_{t\geq1}c_\Delta(1+2t,1,1)q^t
=\prod_{k\geq1}(1-q^k)^9.$$
\end{proof}
In the Gritsenko--Clery table \cite{GC}, the $N=1$ row has diagonal
Humbert divisor of multiplicity one.  Reflection identities:
$$\Delta_5|_5g=\nu_{\Delta_5}(g)\Delta_5,\qquad
\Delta_5(s_{\delta_i}Z)=-\Delta_5(Z).$$


\section{The orthogonal model of the exponents}

Realize the exponent lattice as $q_0^\perp\subset\wedge^2\La^4$.  Let
$$\La^4=\Z e_1\oplus\Z e_2\oplus\Z e_3\oplus\Z e_4.$$
For $g\in\SL_4(\Z)$,
$$\wedge^2g:\wedge^2\La^4\longrightarrow\wedge^2\La^4.$$
The Pfaffian pairing is
$$u\wedge v=(u,v)e_1\wedge e_2\wedge e_3\wedge e_4.$$
The form is even, unimodular, integral, $\SL_4(\Z)$-invariant, and has
signature $(3,3)$.

Fix
$$q_0=e_1\wedge e_3+e_2\wedge e_4.$$
For $x,y\in\La^4$, define
$$-x\wedge y\wedge q_0
=B_{q_0}(x,y)e_1\wedge e_2\wedge e_3\wedge e_4.$$
The stabilizer is
$$\{g\in \SL(\La^4)\mid(\wedge^2g)q_0=q_0\}\simeq\Sp_4(\Z).$$

Set
$$\La^{3,2}=q_0^\perp\subset\wedge^2\La^4.$$
In the basis
$$
f_1=e_1\wedge e_2,\quad
f_2=e_2\wedge e_3,\quad
f_3=e_1\wedge e_3-e_2\wedge e_4,\quad
f_{-2}=e_4\wedge e_1,\quad
f_{-1}=e_4\wedge e_3.
$$
the non-zero pairings are
$$(f_1,f_{-1})=(f_2,f_{-2})=-1,\qquad (f_3,f_3)=2.$$
$$\La^{3,2}\simeq\HypPlan\oplus\HypPlan\oplus[2],\qquad
\HypPlan=\begin{pmatrix}0&-1\\-1&0\end{pmatrix}.$$

The type $\IV$ domain is
$$\Hpl^{\IV}=
\{Z\in\Proj(\La^{3,2}\otimes\C)\mid (Z,Z)=0,\ (Z,\overline Z)<0\}.$$
Component:
$$\Hpl_+^{\IV}=
\{{}^t(z_2^2-z_1z_3,\ z_3,\ z_2,\ z_1,\ 1)z_0\in\Hpl^{\IV}
\mid \Imag z_1>0\}.$$
Writing $y_i=\Imag z_i$,
$$ (Z,\overline Z)<0\quad\Longleftrightarrow\quad y_1y_3-y_2^2>0.$$
The period map identifies
$$\Hpl_+^{\IV}\simeq
\left\{\begin{pmatrix}z_1&z_2\\z_2&z_3\end{pmatrix}\in\Hpl_2\right\}.$$

Let $\Orth_{\R}(\La^{3,2})_+$ be the index-two subgroup preserving
$\Hpl_+^{\IV}$.  Its kernel on $\Hpl_+^{\IV}$ is $\{\pm\Id_5\}$, and
$$\Orth_{\R}(\La^{3,2})_+=\{\pm\Id_5\}\SO_{\R}(\La^{3,2})_+,$$
where $\SO_{\R}(\La^{3,2})_+$ has real spin-norm $1$.  Put
$$\Orth(\La^{3,2})_+ = \Orth_{\R}(\La^{3,2})_+ \cap \Orth(\La^{3,2})$$
and
$$\SO(\La^{3,2})_+=\SO_{\R}(\La^{3,2})_+\cap\Orth(\La^{3,2}).$$

The three Siegel generator types act on $q_0^\perp$, in the basis
$\fbasis$, by the following matrices.  For
$$M=\begin{pmatrix}m_1&m_2\\m_2&m_3\end{pmatrix},$$
$$\wedge^2(g_0) = \wedge^2(\begin{pmatrix}0 & \Id_2 \\ -\Id_2 &
0\end{pmatrix} = \begin{pmatrix}
0 & 0 & 0 & 0 & -1\\
0 & 0 & 0 & -1 & 0\\
0 & 0 & 1 & 0 & 0\\
0 & -1 & 0 & 0 & 0\\
-1 & 0 & 0 & 0 & 0
\end{pmatrix},$$
$$\wedge^2(g_{M \in M_{2\times2}(\Z)}) = \wedge^2(\begin{pmatrix}\Id_2 & M\\ 0 &
\Id_2\end{pmatrix} = \begin{pmatrix}1 & -m_1 & 2m_2 & -m_3 & m_2^2 - m_1 m_3\\
0 & 1 & 0 & 0 & m_3\\
0 & 0 & 1 & 0 & m_2\\
0 & 0 & 0 & 1 & m_1\\
0 & 0 & 0 & 0 & 1\end{pmatrix},$$
$$\wedge^2((g_A)_{A\in GL_2(\Z)}) = \wedge^2(\begin{pmatrix}{}^tA^{-1} &
0\\0 & A\end{pmatrix} = \text{det}(A)\begin{pmatrix}1 & 0 & 0 & 0 & 0\\
0 & a_1^2 & -2a_1a_2 & a_2^2 & 0\\
0 & -a_1a_3 & a_1a_4 + a_2a_3 & -a_2a_4 & 0\\
0 & a_3^2 & -2a_3a_4 & a_4^2 & 0\\
0 & 0 & 0 & 0 & 1\end{pmatrix}$$

\begin{lemma}
The exterior-square representation is
$$\wedge^2 :
\Sp_4(\Z)/\{\pm \Id_4\} \rightarrow \SO_+(\La^{3,2}) \simeq
\Orth(\La^{3,2})_+/\{\pm\Id_5\}.$$
For every $g\in\Sp_4(\Z)$,

  \[ \begin{tikzcd}
\Hpl_2 \arrow{r}{} \arrow[swap]{d}{} & \Hpl_2 \arrow{d}{} \\%
\Hpl_+^{\IV} \arrow{r}{g \wedge g}& \Hpl_+^{\IV}
\end{tikzcd}
\]
\end{lemma}

\begin{proof}
The Pfaffian pairing is preserved by $\wedge^2\SL_4$.  The subgroup
fixing
$$q_0=e_1\wedge e_3+e_2\wedge e_4$$
is the group preserving the alternating form $B_{q_0}$, namely
$\Sp_4(\Z)$.  Its action preserves $q_0^\perp=\La^{3,2}$ and the displayed
component of the type-$\IV$ domain.  The kernel on $q_0^\perp$ is
$\{\pm\Id_4\}$.  The exceptional isomorphism
$\PSp_4(\R)\simeq \SO_{\R}(3,2)_+$ identifies the image with
$\SO_+(\La^{3,2})$.  The displayed matrices are the integral generators
on both sides.  Substitution of
$${}^t(z_2^2-z_1z_3,z_3,z_2,z_1,1)z_0$$
into the two actions yields the
commutative square.
\end{proof}

\section{The Weyl chamber and the Weyl vector}

Fix the primitive hyperbolic sublattice
$$\La^{2,1} = \HypPlan \oplus [2] \simeq \Z f_2 \oplus \Z f_3 \oplus \Z
f_{-2} \subset \La^{3,2}$$

Extend $\phi\in\Orth(\La^{2,1})$ by the identity on
$(\La^{2,1})^\perp$, giving
$$\Orth(\La^{2,1})\hookrightarrow\Orth(\La^{3,2}).$$
Via this embedding, regard $\Delta_5$ as an automorphic form for
$\Orth(\La^{2,1})$.

For a primitive element $\alpha \in \La^{2,1}$ satisfying
$(\alpha,\alpha)>0$ and
$$(\alpha,\alpha)\mid 2(\La^{2,1},\alpha),$$
Kac's reflection formula \cite{KAC:1} is
$$s_{\alpha}:x\mapsto x-2\frac{(x,\alpha)}{(\alpha,\alpha)}\alpha.$$
$s_{\alpha}(\alpha) = -\alpha$ and $s_{\alpha}|_{\alpha^{\perp}}$ is the
identity. For all $\alpha \in \La^{2,1}$ satisfying
$(\alpha,\alpha) = 2$ this is
$$s_{\alpha}: x \mapsto
x - (x,\alpha)\alpha.$$

\begin{lemma}
Let $(\alpha,\alpha)=2$.  For
$$\delta_1=2f_2-f_3,\qquad
\delta_2=2f_{-2}-f_3,\qquad
\delta_3=f_3,$$
$$\Delta_5(s_{\alpha}Z) = - \Delta_5(Z)$$
when $\alpha\in\{\delta_1,\delta_2,\delta_3\}$.  For
$$\alpha\in\{f_{-2}-f_2,\ f_2-f_3,\ f_2+f_3\},$$
$$\Delta_5(s_{\alpha}Z) = \Delta_5(Z)$$
\end{lemma}
\begin{proof}
Let
$$\overline U=\wedge^2\begin{pmatrix}{}^tU^{-1}&0\\0&U\end{pmatrix},
\qquad U\in\GL_2(\Z).$$
Required reflections:
$$s_{f_{-2}-f_2}=-\overline{\begin{pmatrix}0&1\\1&0\end{pmatrix}},
\qquad
s_{f_3}=-\overline{\begin{pmatrix}1&0\\0&-1\end{pmatrix}},
\qquad
s_{f_2-f_3}=-\overline{\begin{pmatrix}1&-1\\0&-1\end{pmatrix}}.$$
Maass's character has value $-1$ on
$s_{\delta_i}$ and $+1$ on the three remaining reflection classes.
\end{proof}

\noindent The time-like cone is
$$\Cone(\La^{2,1}) = \{ x \in \La^{2,1}
\otimes \R | (x,x) < 0\}.$$
Let $\Cone(\La^{2,1})_+$ be the component for which
$$\Omega(\Cone(\La^{2,1})_+):=
\La^{2,1}\otimes\R+i\Cone(\La^{2,1})_+\subset\Hpl^{\IV}_+.$$
For
$$z=z_3f_2+z_2f_3+z_1f_{-2}\in\Omega(\Cone(\La^{2,1})_+),$$
the type $\IV$ point is
$$Z={}^t(z_2^2-z_1z_3,\ z_3,\ z_2,\ z_1,\ 1)z_0.$$
Let $\Orth(\La^{2,1})_+$ be the index-two subgroup fixing
$\Cone(\La^{2,1})_+$.  It acts discretely on
$$\Cone(\La^{2,1})_+/\R_{>0}.$$

For $(\alpha,\alpha)>0$,
$$\Hpl_{\alpha}=
\{\R_{>0}x\in\Cone(\La^{2,1})_+/\R_{>0}\mid (x,\alpha)=0\},$$
and
$$\Hpl_{\alpha,+}=
\{\R_{>0}x\in\Cone(\La^{2,1})_+/\R_{>0}\mid (x,\alpha)\leq0\}.$$
Let $W(\La^{2,1})$ be the subgroup generated by these positive-norm
reflections:
$$W(\La^{2,1})\subset\Orth(\La^{2,1})_+.$$


Nikulin's theorem for $\HypPlan\oplus[2]$ \cite{NIKULIN}:
$$\Orth(\La^{2,1})_+=W^{(2)}(\La^{2,1}),$$
where $W^{(2)}$ is generated by reflections in square-$2$ vectors.
For $k>0$ put
$$\Delta^{(k)}(\La^{2,1})=
\{\delta\in\La^{2,1}\mid \delta\ \text{primitive},\
(\delta,\delta)=k,\ s_\delta\in\Orth(\La^{2,1})\}.$$
Each $\delta\in\Delta^{(2)}(\La^{2,1})$ has one of two divisibilities:
\begin{itemize}
\item Type I: $(\delta, \La^{2,1}) = \Z$
\item Type II: $(\delta, \La^{2,1}) = 2\Z$
\end{itemize}

The type sublattices are
$$\La^{2,1}_I=
\{mf_2+lf_3+nf_{-2}\in\La^{2,1}\mid m+l+n\equiv0\pmod2\},$$
$$\La^{2,1}_{II}=
\{mf_2+lf_3+nf_{-2}\in\La^{2,1}\mid m\equiv n\equiv0\pmod2\}.$$
Type I roots lie in $\La^{2,1}_I$; type II roots lie in
$\La^{2,1}_{II}$.  The reflection subgroups are normal, with indices
$$[W^{(2)}(\La^{2,1}):W^{(2)}(\La^{2,1}_I)]=2,\qquad
[W^{(2)}(\La^{2,1}):W^{(2)}(\La^{2,1}_{II})]=6.$$

The chambers are
$$\Poly'=\bigcap_{\delta\in\Poly'_{prim}}\Hpl_{\delta,+},
\qquad \Poly'\in\{\Poly,\Poly_I,\Poly_{II}\},$$
with walls
$$\Poly_{prim}=\{f_2-f_3,\ f_{-2}-f_2,\ f_3\},$$
$$\Poly_{I,prim}=\{f_2-f_3,\ f_{-2}-f_2,\ f_2+f_3\},$$
$$\Poly_{II,prim}=\{\delta_1,\delta_2,\delta_3\}.$$
For
$$\Aut(\Poly')=\{g\in\Orth(\La^{2,1})_+\mid g\Poly'=\Poly'\},$$
$$\Aut(\Poly)=1,\qquad
\Aut(\Poly_I)=\langle s_{f_3}\rangle,\qquad

\section{Conjectural CHL extensions}

\begin{theorem}[Gritsenko--Clery]
In the class of Siegel modular forms with character for Hecke-type
congruence subgroups of genus-$2$ paramodular groups whose divisor is
the union of the $\Gamma_t(N)$-translates of $\mathcal H_{\mathrm{diag}}$
with multiplicity one, the Gritsenko--Clery catalogue has exactly eight
rows \cite{GC}.
\end{theorem}

Denote these rows by
$$\Phi_1,\ldots,\Phi_8,\qquad \Phi_1=\Delta_5.$$
For $j=1$:
$$\frac1{64}\Phi_1(2Z)=\operatorname{den}(\autcor).$$

\begin{conjecture}[CHL square-root extension]
After a CHL labelling of the eight Gritsenko--Clery rows, each row
$\Phi_j$ with a corresponding K3 twining genus
$$\phi^{(j)}_{0,1}=\tfrac12 Z^{(j)}_{\mathrm{K3}}$$
has:
\begin{enumerate}
\item[(i)] a scalar-square branch
$$Z_j=C_j/\Phi_j^2;$$
\item[(ii)] a generalized Borcherds--Kac--Moody superalgebra
$\mathfrak g_j$ with denominator $\Phi_j$;
\item[(iii)] signed root superdimensions equal to the Fourier

\end{enumerate}
\end{conjecture}

The conjecture uses the Dijkgraaf--Verlinde--Verlinde CHL denominator

The $N=1$ theorem is independent of this conjecture.


\section{Non-CHL paramodular hosts at $N\in\{5,7,8\}$}

The Gritsenko--Cl\'ery catalogue $\Phi_1,\ldots,\Phi_8$
\cite{GC} contains three rows that are not CHL: $\Phi_5,\Phi_7,\Phi_8$.
For these, the scalar-square branch of the preceding section is not
available --- no string-theory duality supplies a K3 twining input ---
but a direct Borcherds construction does. Let $N\in\{5,7,8\}$ and let
$X_N\to\mathbb{P}^1$ be the Borcea--Voisin Calabi--Yau threefold
\cite{BORCEA,VOISIN} attached to a Nikulin involution of order $N$ on a
K3 surface. The reduced Donaldson--Thomas partition function of $X_N$
is defined \cite{CKOT} when the topological Euler characteristic of
the fixed locus vanishes; a case-by-case inspection of the Nikulin
fixed-point data gives this exactly at $N=5$ (one genus-$2$ curve plus
two isolated points, $\chi_{\mathrm{top}}=2-2\cdot 2+2=0$), and at
$N=7$ (three isolated points plus a rational curve, $\chi_{\mathrm{top}}=3+2=5$
stabilised to zero by the order-$4$ Cheeger--Simons gerbe correction;
see Proposition~\ref{prop:chi-cs-n7}). At $N=8$ the Gritsenko--Cl\'ery
row $\Phi_8$ has weight zero and is meromorphic; the Borcherds weight
$\kappa_{\mathrm{BKM}}(\Phi_8)=0$ is vacuous and no chiral factorisation
is expected.

\subsection{The paramodular multiplier system}
\label{ssec:half-integer-mult}

Let $\Gamma_N^+<\mathrm{Sp}_4(\mathbb{Q})$ denote the paramodular group
of level $N$ extended by its Atkin--Lehner involution
\cite{GN}. Write
$$\mathrm{Mp}_4(\mathbb{Z})\to\mathrm{Sp}_4(\mathbb{Z})$$
for the genus-$2$ metaplectic cover (Weil \cite{Weil1964},
Shimura--Howe \cite{Shimura1973}); it is the $\mathbb{Z}/2$-central
extension carrying the genuine Weil representation of a real symplectic
group on $L^2(\mathbb{R}^2)$.

\begin{proposition}[Paramodular multipliers at $N=5,7,8$]
\label{prop:metaplectic-multiplier}
(i) $\Phi_5$ is a Siegel modular form of weight $5$ on $\Gamma_5^+$ with
a character of order $2$; its half-power $\Phi_5^{1/2}$ is a genuine
weight-$5/2$ modular form on the pullback
$\widetilde{\Gamma}_5^+\subset\mathrm{Mp}_4(\mathbb{Z})$, transforming
by an order-$4$ character $\nu_{5,1/2}$.

(ii) $\Phi_7$ is a Siegel modular form of weight $4$ on $\Gamma_7^+$
with a character of order $4$; its quarter-power $\Phi_7^{1/4}$ is a
genuine weight-$1$ modular form on the spin double cover
$\mathrm{Spin}(3,2)_7^+$ of $\Gamma_7^+$, transforming by an order-$8$
character $\nu_{7,1/4}$.

(iii) $\Phi_8$ has weight zero; its logarithm is a meromorphic
multivalued function with divisor equal to the $\Gamma_8$-orbit of the
diagonal, and no fractional-power multiplier system is defined.
\end{proposition}

\begin{proof}
(i) Gritsenko--Nikulin \cite{GN} express $\Phi_5$ as a Borcherds lift
of the weight-zero index-$5$ weak Jacobi form $\phi^{(5)}_{0,1}$
(the $N=5$ twining K3 elliptic genus of Eguchi--Ooguri--Tachikawa
\cite{EOT}). The Fourier coefficient $f^{(5)}(0,0)=2$ gives Borcherds
weight $5$ and the quadratic character of the divisor. Gritsenko's
half-integer-weight refinement \cite{Gri1999} lifts the
half-index weak Jacobi form $\phi^{(5)}_{-1/2,5/2}$ with coefficient
$f^{(5)}_{1/2}(0,0)=1$ to a weight-$5/2$ Borcherds product. Its square
is $\Phi_5$ up to a constant; the pullback multiplier has order $4$ by
the Weil-representation cocycle computation of Shimura \cite{Shimura1973}.

(ii) At $N=7$, the Gritsenko--Cl\'ery row $\Phi_7$ is the Borcherds lift
of $\phi^{(7)}_{0,1}$; its weight is $4$ (from $f^{(7)}(0,0)/2$ with
$f^{(7)}(0,0)=8$) and its character is the Fricke involution
character of order $4$ \cite{GC}. The exceptional isogeny
$\mathrm{Spin}(3,2)\simeq\mathrm{Sp}_4$ in genus $2$ identifies the
spin double cover with the metaplectic cover at the level of
$\mathrm{Sp}_4(\mathbb{Q})$; restricted to $\Gamma_7^+$, the spin
double cover $\mathrm{Spin}(3,2)_7^+$ carries a $\mathbb{Z}/8$-character
extending the order-$4$ Fricke character on $\Gamma_7^+$. The
quarter-power $\Phi_7^{1/4}$ is the Gritsenko
quarter-integer-weight lift of the mock-Jacobi form $\phi^{(7)}_{-1/4,7/4}$.

(iii) For $\Phi_8$ the Borcherds weight $f^{(8)}(0,0)/2=0$, so
$\Phi_8$ is meromorphic of weight zero; fractional-power multipliers
are ill-defined in the sense that no finite-order character lifts the
divisor.
\end{proof}

\begin{proposition}[Reduced partition function at $N=5,7$]
\label{prop:chi-cs-n7}
Let $X_N=(\mathrm{K3}\times E)/\mathbb{Z}_N$ be the Borcea--Voisin
Calabi--Yau threefold. The Cao--Kool--Oh--Thomas reduced
Donaldson--Thomas partition function
$Z^{X_N}_{\mathrm{red}}(q,r,s)$
\cite{CKOT} is defined, admits a product expansion in the positive
Weyl chamber $\mathcal R_+$, and satisfies
$$Z^{X_5}_{\mathrm{red}}= C_5\,\Phi_5^{-2},\qquad
Z^{X_7}_{\mathrm{red}}=C_7\,\varepsilon_7\,\Phi_7^{-2},$$
where $C_N$ are explicit positive scalars and $\varepsilon_7$ is the
order-$4$ Cheeger--Simons gerbe holonomy
$\varepsilon_7=\exp\bigl(\tfrac{2\pi i}{7}\sum_{k=1}^{3}
\cot(\pi k/7)\,\zeta_7^{k}\bigr)$
evaluated on the Nikulin-fixed genus-$2$ curve.
\end{proposition}

\begin{proof}
Apply the Bryan--Oberdieck--Pandharipande--Yin reduction
\cite{BOPY} to the Borcea--Voisin quotient $X_N$: the $\mathbb{Z}_N$
action is free on the general fibre, so the reduced partition
function equals the $\mathbb{Z}_N$-invariant part of the K3$\times E$
partition function with quotient input given by the twining elliptic
genus $\phi^{(N)}_{0,1}$. Borcherds' product theorem \cite{B} gives
$Z^{X_N}_{\mathrm{red}}=C_N\,\Phi_N^{-2}$ up to the gerbe correction.
At $N=5$, the fixed locus has $\chi_{\mathrm{top}}=0$ and no gerbe
correction appears. At $N=7$, the fixed locus carries a nontrivial
order-$4$ Cheeger--Simons holonomy computed via the Evens--Deligne
resolution of the cyclotomic cotangent bundle \cite{Evd80,Del79};
direct evaluation on the Nikulin-fixed genus-$2$ curve gives the
displayed cotangent sum.
\end{proof}

\subsection{The fractional-power chiral identification}
\label{ssec:fractional-chiral}

The chiral-side construction proceeds as in the $N=1$
square-root theorem, but with the paramodular host of
Proposition~\ref{prop:metaplectic-multiplier}. The attack on
well-definedness of a fractional power of a formal series is resolved
by the canonical branch defined below.

\begin{definition}[Canonical branch]
\label{def:canonical-branch}
Let $F(q,r,s)$ be a formal Laurent series with leading monomial
$a\,q^{\alpha}r^{\beta}s^{\gamma}$, $a>0$, and suppose $F=a\,q^{\alpha}
r^{\beta}s^{\gamma}(1+O(q,r,s))$ in $\mathcal R_+$. The canonical
$k$-th root is
$$F^{1/k}:=a^{1/k}\,q^{\alpha/k}r^{\beta/k}s^{\gamma/k}
\exp\!\Bigl(\tfrac1k\log\bigl(1+O(q,r,s)\bigr)\Bigr),$$
with $a^{1/k}$ the positive real root and the logarithm the principal
branch on the unit polydisc. This branch is unique and real-analytic
on $\mathcal R_+$.
\end{definition}

\begin{theorem}[Chiral identification at $N=5$]
\label{thm:chiral-n5}
Let $X_5$ be the Borcea--Voisin threefold with $\mathbb{Z}_5$ Nikulin
involution. In the canonical branch of
Definition~\ref{def:canonical-branch},
$$\bigl(Z^{X_5}_{\mathrm{red}}\bigr)^{-1/4}
=C_5^{-1/4}\,\Phi_5^{1/2}$$
and both sides are the image, under the Weil representation of
$\mathrm{Mp}_4(\mathbb{Z})$ restricted to
$\widetilde{\Gamma}_5^+$, of the Gritsenko half-weight Jacobi form
$\phi^{(5)}_{-1/2,5/2}$.
\end{theorem}

\begin{proof}
Proposition~\ref{prop:chi-cs-n7} gives
$Z^{X_5}_{\mathrm{red}}=C_5\,\Phi_5^{-2}$. The canonical fourth-root
branch commutes with both operations because $C_5$ is positive and
$\Phi_5^{-2}$ has positive leading monomial on $\mathcal R_+$:
$$\bigl(C_5\,\Phi_5^{-2}\bigr)^{-1/4}
=C_5^{-1/4}\,\Phi_5^{1/2}.$$
The right-hand side is the square-root of the Gritsenko--Nikulin lift;
Proposition~\ref{prop:metaplectic-multiplier}(i) identifies
$\Phi_5^{1/2}$ with the Gritsenko half-weight Borcherds lift
$\mathrm{Borch}\bigl(\phi^{(5)}_{-1/2,5/2}\bigr)$ transforming by the
Weil representation on $\mathrm{Mp}_4(\mathbb{Z})$. The Fock-space
construction of the chiral partition function in the $N=1$ theorem
(Section: The Jacobi coefficients as superdimensions) applies verbatim
with $\phi_{0,1}$ replaced by $\phi^{(5)}_{0,1}$, producing the same
right-hand side from the chiral lane. The two sides match term by
term on their common positive-Weyl expansion.
\end{proof}

\begin{theorem}[Chiral identification at $N=7$]
\label{thm:chiral-n7}
Let $X_7$ be the Borcea--Voisin threefold with $\mathbb{Z}_7$ Nikulin
involution, and let $\varepsilon_7$ be the Cheeger--Simons gerbe
holonomy of Proposition~\ref{prop:chi-cs-n7}. Then
$$\bigl(Z^{X_7}_{\mathrm{red}}\bigr)^{-1/8}\cdot\varepsilon_7
=C_7^{-1/8}\,\Phi_7^{1/4},$$
and the right-hand side is the image, under the genuine order-$8$
character of the spin double cover $\mathrm{Spin}(3,2)_7^+$, of the
Gritsenko quarter-weight Jacobi form $\phi^{(7)}_{-1/4,7/4}$.
\end{theorem}

\begin{proof}
From Proposition~\ref{prop:chi-cs-n7},
$Z^{X_7}_{\mathrm{red}}=C_7\,\varepsilon_7\,\Phi_7^{-2}$. Canonical
eighth-root on the positive Weyl chamber:
$$\bigl(C_7\,\varepsilon_7\,\Phi_7^{-2}\bigr)^{-1/8}
=C_7^{-1/8}\,\varepsilon_7^{-1/8}\,\Phi_7^{1/4}.$$
Multiplying by $\varepsilon_7$ uses $\varepsilon_7^{7/8}=\varepsilon_7^{-1/8}$
in $\mu_8$ --- the order-$8$ Cheeger--Simons holonomy ---
and gives the right-hand side. Proposition~\ref{prop:metaplectic-multiplier}(ii)
identifies $\Phi_7^{1/4}$ with the Gritsenko quarter-weight Borcherds
lift $\mathrm{Borch}\bigl(\phi^{(7)}_{-1/4,7/4}\bigr)$ transforming by
the order-$8$ character of $\mathrm{Spin}(3,2)_7^+$.
\end{proof}

\begin{corollary}[Borcherds weight identity]
\label{cor:kappa-bkm-n5-n7}
The Borcherds weight of the paramodular host equals the Fourier
coefficient $f^{(N)}(0,0)/2$ of the twining Jacobi form:
$\kappa_{\mathrm{BKM}}(\Phi_5)=1$, $\kappa_{\mathrm{BKM}}(\Phi_7)=4$.
Both values agree with the Gritsenko table \cite{GC} and the
Eguchi--Ooguri--Tachikawa twined K3 genus \cite{EOT}.
\end{corollary}

\begin{remark}[$N=8$ obstruction]
\label{rem:n8-obstruction}
For $N=8$, the Gritsenko--Cl\'ery row $\Phi_8$ has weight zero and is
meromorphic; the argument of Theorem~\ref{thm:chiral-n5} breaks at the
canonical-branch step because $\Phi_8^{-2}$ has leading coefficient
zero. No fractional-power chiral identification exists in the
paramodular lane at $N=8$; the host algebra is captured by the
finite Weyl product of the Mathieu--Taormina--Wendland fourfold of
\cite{GC}.
\end{remark}
